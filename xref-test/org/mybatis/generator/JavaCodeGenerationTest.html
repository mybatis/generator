16  package org.mybatis.generator;
17  
18  import static org.junit.jupiter.api.Assertions.fail;
19  
20  import com.github.javaparser.JavaParser;
21  import com.github.javaparser.ParseProblemException;
22  import com.github.javaparser.ParserConfiguration;
23  
24  import java.io.InputStream;
25  import java.util.ArrayList;
26  import java.util.List;
27  
28  import org.junit.jupiter.params.ParameterizedTest;
29  import org.junit.jupiter.params.provider.MethodSource;
30  import org.mybatis.generator.api.GeneratedJavaFile;
31  import org.mybatis.generator.api.MyBatisGenerator;
32  import org.mybatis.generator.api.dom.DefaultJavaFormatter;
33  import org.mybatis.generator.config.Configuration;
34  import org.mybatis.generator.config.xml.ConfigurationParser;
35  import org.mybatis.generator.internal.DefaultShellCallback;
36  
37  class JavaCodeGenerationTest {
38  
39      @ParameterizedTest
40      @MethodSource("javaFileGenerator")
41      void testJavaParse(GeneratedJavaFile generatedJavaFile) {
42          DefaultJavaFormatter formatter = new DefaultJavaFormatter();
43  
44          String content = formatter.getFormattedContent(generatedJavaFile.getCompilationUnit());
45          try {
46              ParserConfiguration parserConfiguration = new ParserConfiguration();
47              parserConfiguration.setLanguageLevel(ParserConfiguration.LanguageLevel.JAVA_25);
48              JavaParser javaParser = new JavaParser(parserConfiguration);
49              javaParser.parse(content);
50          } catch (ParseProblemException e) {
51              fail("Generated Java File " + generatedJavaFile.getFileName() + " will not compile");
52          }
53      }
54  
55      static List<GeneratedJavaFile> javaFileGenerator() throws Exception {
56          List<GeneratedJavaFile> generatedFiles = new ArrayList<>();
57          generatedFiles.addAll(generateJavaFilesMybatis());
58          generatedFiles.addAll(generateJavaFilesMybatisDsql());
59          return generatedFiles;
60      }
61  
62      static List<GeneratedJavaFile> generateJavaFilesMybatis() throws Exception {
63          createDatabase();
64          return generateJavaFiles("/scripts/generatorConfig.xml");
65      }
66  
67      static List<GeneratedJavaFile> generateJavaFilesMybatisDsql() throws Exception {
68          createDatabase();
69          return generateJavaFiles("/scripts/generatorConfig_Dsql.xml");
70      }
71  
72      static List<GeneratedJavaFile> generateJavaFiles(String configFile) throws Exception {
73          ConfigurationParser cp = new ConfigurationParser();
74          InputStream is = JavaCodeGenerationTest.class.getResourceAsStream(configFile);
75          if (is == null) {
76              throw new RuntimeException("Cannot load configFile: " + configFile);
77          }
78          Configuration config = cp.parseConfiguration(is);
79  
80          MyBatisGenerator myBatisGenerator = new MyBatisGenerator.Builder()
81                  .withConfiguration(config)
82                  .withShellCallback(new DefaultShellCallback())
83                  .withOverwriteEnabled(true)
84                  .build();
85          myBatisGenerator.generateOnly();
86          return myBatisGenerator.getGeneratedJavaFiles();
87      }
88  
89      static void createDatabase() throws Exception {
90          SqlScriptRunner scriptRunner =
91                  new SqlScriptRunner(JavaCodeGenerationTest.class.getResourceAsStream("/scripts/CreateDB.sql"),
92                          "org.hsqldb.jdbcDriver", "jdbc:hsqldb:mem:aname", "sa", "");
93          scriptRunner.executeScript();
94      }
95  }
