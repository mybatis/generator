18  import static org.junit.jupiter.api.Assertions.*;
19  
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.junit.jupiter.params.ParameterizedTest;
24  import org.junit.jupiter.params.provider.MethodSource;
25  import org.mybatis.generator.api.GeneratedKotlinFile;
26  import org.mybatis.generator.api.MyBatisGenerator;
27  import org.mybatis.generator.config.Configuration;
28  import org.mybatis.generator.config.xml.ConfigurationParser;
29  import org.mybatis.generator.internal.DefaultShellCallback;
30  
31  /**
32   * This test executes the same generator configuration that is in the mybatis-generator-systests-kotlin
33   * project. The purpose of this test is to exercise the Kotlin code generators during the initial build
34   * as a kind of smoke test - currently we do not try to parse the generated code. This test also shows
35   * test coverage as it mimics the code paths executed by the integration test.
36   *
37   * @author Jeff Butler
38   *
39   */
40  class KotlinCodeGenerationTest {
41  
42      @ParameterizedTest
43      @MethodSource("kotlinFileGenerator")
44      void testKotlinParse(GeneratedKotlinFile generatedKotlinFile) {
45          // for now, just let the test pass. if we find a good Kotlin parser, then mimic the
46          // function of the Java test by trying to parse generated code
47          assertTrue(true);
48      }
49  
50      static List<GeneratedKotlinFile> kotlinFileGenerator() throws Exception {
51          JavaCodeGenerationTest.createDatabase();
52          return generateKotlinFiles("/scripts/generatorConfig-kotlin.xml");
53      }
54  
55      static List<GeneratedKotlinFile> generateKotlinFiles(String configFile) throws Exception {
56          List<String> warnings = new ArrayList<>();
57          ConfigurationParser cp = new ConfigurationParser(warnings);
58          Configuration config = cp.parseConfiguration(KotlinCodeGenerationTest.class.getResourceAsStream(configFile));
59  
60          DefaultShellCallback shellCallback = new DefaultShellCallback(true);
61  
62          MyBatisGenerator myBatisGenerator = new MyBatisGenerator(config, shellCallback, warnings);
63          myBatisGenerator.generate(null, null, null, false);
64          return myBatisGenerator.getGeneratedKotlinFiles();
65      }
66  }
