16  package org.mybatis.generator.api.dom.java;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  /**
22   * This test verifies the ability of the generator to use fully qualified names
23   * in code generation when the type is not explicitly imported.
24   *
25   * <p>The test generates a hierarchy with multiple classes that have the same name in
26   * different packages.
27   */
28  public class ComplexHierarchyGenerator {
29  
30      private static final String BASE_PACKAGE = "mbg.domtest.generators.fieldtype1.output";
31  
32      public static List<CompilationUnit> generateTestClasses() {
33          FullyQualifiedJavaType cls = new FullyQualifiedJavaType(BASE_PACKAGE + ".SomeClass");
34  
35          List<CompilationUnit> answer = new ArrayList<>();
36          TopLevelClass tlcMain = generateFieldTypeMain();
37          TopLevelClass tlcSub1 = generateFieldTypeSub1();
38          TopLevelClass tlcTcSub1 = generateTestClassSub1();
39          TopLevelClass tlcSub2 = generateFieldTypeSub2();
40  
41          answer.add(tlcMain);
42          answer.add(tlcSub1);
43          answer.add(tlcTcSub1);
44          answer.add(tlcSub2);
45  
46          TopLevelClass topLvlClass = new TopLevelClass(cls);
47          topLvlClass.setVisibility(JavaVisibility.PUBLIC);
48          topLvlClass.addImportedType(tlcTcSub1.getType());
49  
50          Field field = new Field("main", tlcMain.getType());
51          field.setVisibility(JavaVisibility.PRIVATE);
52          topLvlClass.addField(field);
53  
54          field = new Field("tcSub1", tlcTcSub1.getType());
55          field.setVisibility(JavaVisibility.PRIVATE);
56          topLvlClass.addField(field);
57  
58          field = new Field("sub1", tlcSub1.getType());
59          field.setVisibility(JavaVisibility.PRIVATE);
60          topLvlClass.addField(field);
61  
62          field = new Field("sub2", tlcSub2.getType());
63          field.setVisibility(JavaVisibility.PRIVATE);
64          topLvlClass.addField(field);
65  
66          Method m = new Method("executeMain");
67          m.setVisibility(JavaVisibility.PUBLIC);
68          m.addBodyLine("main.mainMethod();");
69          topLvlClass.addMethod(m);
70  
71          m = new Method("setMain");
72          m.setVisibility(JavaVisibility.PUBLIC);
73          m.addParameter(new Parameter(tlcMain.getType(), "main"));
74          m.addBodyLine("this.main = main;");
75          topLvlClass.addMethod(m);
76  
77          m = new Method("getMain");
78          m.setVisibility(JavaVisibility.PUBLIC);
79          m.setReturnType(tlcMain.getType());
80          m.addBodyLine("return main;");
81          topLvlClass.addMethod(m);
82  
83          m = new Method("executeSub1");
84          m.setVisibility(JavaVisibility.PUBLIC);
85          m.addBodyLine("sub1.sub1Method();");
86          topLvlClass.addMethod(m);
87  
88          m = new Method("setSub1");
89          m.setVisibility(JavaVisibility.PUBLIC);
90          m.addParameter(new Parameter(tlcSub1.getType(), "sub1"));
91          m.addBodyLine("this.sub1 = sub1;");
92          topLvlClass.addMethod(m);
93  
94          m = new Method("getSub1");
95          m.setVisibility(JavaVisibility.PUBLIC);
96          m.setReturnType(tlcSub1.getType());
97          m.addBodyLine("return sub1;");
98          topLvlClass.addMethod(m);
99  
100         m = new Method("executeSub2");
101         m.setVisibility(JavaVisibility.PUBLIC);
102         m.addBodyLine("sub2.sub2Method();");
103         topLvlClass.addMethod(m);
104 
105         m = new Method("setSub2");
106         m.setVisibility(JavaVisibility.PUBLIC);
107         m.addParameter(new Parameter(tlcSub2.getType(), "sub2"));
108         m.addBodyLine("this.sub2 = sub2;");
109         topLvlClass.addMethod(m);
110 
111         m = new Method("getSub2");
112         m.setVisibility(JavaVisibility.PUBLIC);
113         m.setReturnType(tlcSub2.getType());
114         m.addBodyLine("return sub2;");
115         topLvlClass.addMethod(m);
116 
117         answer.add(topLvlClass);
118 
119         return answer;
120     }
121 
122     private static TopLevelClass generateFieldTypeMain() {
123         FullyQualifiedJavaType fqjt = new FullyQualifiedJavaType(BASE_PACKAGE + ".FieldType");
124         TopLevelClass tlc = new TopLevelClass(fqjt);
125         tlc.setVisibility(JavaVisibility.PUBLIC);
126 
127         Method m = new Method("mainMethod");
128         m.setVisibility(JavaVisibility.PUBLIC);
129         m.addBodyLine("System.out.println(\"main method\");");
130         tlc.addMethod(m);
131 
132         return tlc;
133     }
134 
135     private static TopLevelClass generateFieldTypeSub1() {
136         FullyQualifiedJavaType fqjt = new FullyQualifiedJavaType(BASE_PACKAGE + ".sub1.FieldType");
137         TopLevelClass tlc = new TopLevelClass(fqjt);
138         tlc.setVisibility(JavaVisibility.PUBLIC);
139 
140         Method m = new Method("sub1Method");
141         m.setVisibility(JavaVisibility.PUBLIC);
142         m.addBodyLine("System.out.println(\"sub1 method\");");
143         tlc.addMethod(m);
144 
145         return tlc;
146     }
147 
148     private static TopLevelClass generateTestClassSub1() {
149         FullyQualifiedJavaType fqjt = new FullyQualifiedJavaType(BASE_PACKAGE + ".sub1.SomeClass");
150         TopLevelClass tlc = new TopLevelClass(fqjt);
151         tlc.setVisibility(JavaVisibility.PUBLIC);
152 
153         Method m = new Method("testClassMethod");
154         m.setVisibility(JavaVisibility.PUBLIC);
155         m.addBodyLine("System.out.println(\"testClass sub1 method\");");
156         tlc.addMethod(m);
157 
158         return tlc;
159     }
160 
161     private static TopLevelClass generateFieldTypeSub2() {
162         FullyQualifiedJavaType fqjt = new FullyQualifiedJavaType(BASE_PACKAGE + ".sub2.FieldType");
163         TopLevelClass tlc = new TopLevelClass(fqjt);
164         tlc.setVisibility(JavaVisibility.PUBLIC);
165 
166         Method m = new Method("sub2Method");
167         m.setVisibility(JavaVisibility.PUBLIC);
168         m.addBodyLine("System.out.println(\"sub2 method\");");
169         tlc.addMethod(m);
170 
171         return tlc;
172     }
173 }
