22  public class InnerClassTest {

24      private static final String LF = System.getProperty("line.separator");

26      @Test
27      public void testConstructor() {
28          InnerClass clazz = new InnerClass("com.foo.UserClass");
29  
30          assertNotNull(clazz);
31  
32          assertNull(clazz.getSuperClass());
33  
34          assertNotNull(clazz.getType());
35          assertEquals("com.foo.UserClass", clazz.getType().getFullyQualifiedName());
36  
37          assertNotNull(clazz.getFields());
38          assertEquals(0, clazz.getFields().size());
39  
40          assertNotNull(clazz.getInnerClasses());
41          assertEquals(0, clazz.getInnerClasses().size());
42  
43          assertNotNull(clazz.getInnerEnums());
44          assertEquals(0, clazz.getInnerEnums().size());
45  
46          assertNotNull(clazz.getTypeParameters());
47          assertEquals(0, clazz.getTypeParameters().size());
48  
49          assertNotNull(clazz.getSuperInterfaceTypes());
50          assertEquals(0, clazz.getSuperInterfaceTypes().size());
51  
52          assertNotNull(clazz.getMethods());
53          assertEquals(0, clazz.getMethods().size());
54  
55          assertNotNull(clazz.getInitializationBlocks());
56          assertEquals(0, clazz.getInitializationBlocks().size());
57  
58          assertFalse(clazz.isAbstract());
59      }

61      @Test
62      public void testAddFields() {
63          InnerClass clazz = new InnerClass("com.foo.UserClass");
64  
65          assertEquals(0, clazz.getFields().size());

68          clazz.addField(new Field("test2", FullyQualifiedJavaType.getStringInstance()));

157     @Test
158     public void testGetFormattedContent() {
159         InnerClass clazz = new InnerClass("com.foo.UserClass");
160         clazz.addField(new Field("test", FullyQualifiedJavaType.getStringInstance()));
161         clazz.setSuperClass("com.hoge.SuperClass");
162         clazz.addInnerClass(new InnerClass("InnerUserClass"));
163         clazz.addInnerEnum(new InnerEnum(new FullyQualifiedJavaType("TestEnum")));
164         clazz.addTypeParameter(new TypeParameter("T"));
165         clazz.addTypeParameter(new TypeParameter("U"));
166         clazz.addInitializationBlock(new InitializationBlock(false));
167         clazz.addSuperInterface(new FullyQualifiedJavaType("com.hoge.UserInterface"));
168         clazz.addMethod(new Method("method1"));
169         clazz.setAbstract(true);
170 
171         String excepted = "abstract class UserClass<T, U>  extends SuperClass implements UserInterface {" + LF
172                 + "    String test;" + LF
173                 + "" + LF
174                 + "    {" + LF
175                 + "    }" + LF
176                 + "" + LF
177                 + "    abstract void method1();" + LF
178                 + "" + LF
179                 + "    class InnerUserClass {" + LF
180                 + "    }" + LF
181                 + "" + LF
182                 + "    enum TestEnum {" + LF
183                 + "    }" + LF
184                 + "}";
185 
186         assertEquals(excepted, clazz.getFormattedContent(0, null));
187     }
