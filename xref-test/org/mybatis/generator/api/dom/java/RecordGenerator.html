16  package org.mybatis.generator.api.dom.java;
17  
18  import java.util.List;
19  
20  public class RecordGenerator {
21  
22      public static List<CompilationUnit> generateTestClasses() {
23          return List.of(simpleRecord(), complexRecord());
24      }
25  
26      private static CompilationUnit simpleRecord() {
27          TopLevelRecord topLevelRecord = new TopLevelRecord("mbg.domtest.generators.records.SimpleRecord");
28          topLevelRecord.setVisibility(JavaVisibility.PUBLIC);
29          topLevelRecord.addParameter(new Parameter(new FullyQualifiedJavaType("int"), "id"));
30          topLevelRecord.addParameter(new Parameter(new FullyQualifiedJavaType("String"), "description"));
31  
32          return topLevelRecord;
33      }
34  
35      public static CompilationUnit complexRecord() {
36          FullyQualifiedJavaType recordType = new FullyQualifiedJavaType("mbg.domtest.generators.records.ComplexRecord");
37          TopLevelRecord topLevelRecord = new TopLevelRecord(recordType);
38          topLevelRecord.addTypeParameter(new TypeParameter("T"));
39          topLevelRecord.addSuperInterface(new FullyQualifiedJavaType("java.io.Serializable"));
40          topLevelRecord.addImportedType("java.io.Serializable");
41          topLevelRecord.setVisibility(JavaVisibility.PUBLIC);
42          topLevelRecord.addParameter(new Parameter(new FullyQualifiedJavaType("int"), "id"));
43          topLevelRecord.addParameter(new Parameter(new FullyQualifiedJavaType("String"), "description"));
44          topLevelRecord.addParameter(new Parameter(new FullyQualifiedJavaType("T"), "theT"));
45          topLevelRecord.addFileCommentLine("/**");
46          topLevelRecord.addFileCommentLine(" * A complex record");
47          topLevelRecord.addFileCommentLine(" */");
48          topLevelRecord.addStaticImport("java.lang.Math.PI");
49  
50          Field field = new Field("f", new FullyQualifiedJavaType("int"));
51          field.setFinal(true);
52          field.setStatic(true);
53          topLevelRecord.addField(field);
54  
55          InitializationBlock ib = new InitializationBlock(true);
56          ib.addBodyLine("f = 22;");
57          topLevelRecord.addInitializationBlock(ib);
58  
59          Method constructor = new Method("ComplexRecord");
60          constructor.setConstructor(true);
61          constructor.addParameter(new Parameter(new FullyQualifiedJavaType("int"), "id"));
62          constructor.addParameter(new Parameter(new FullyQualifiedJavaType("String"), "description"));
63          constructor.addBodyLine("this(id, description, null);");
64          topLevelRecord.addMethod(constructor);
65  
66          Method method = new Method("ff");
67          method.setStatic(true);
68          method.setReturnType(FullyQualifiedJavaType.getIntInstance());
69          method.addBodyLine("return f;");
70          topLevelRecord.addMethod(method);
71  
72          method = new Method("add");
73          method.setReturnType(FullyQualifiedJavaType.getIntInstance());
74          method.addParameter(new Parameter(FullyQualifiedJavaType.getIntInstance(), "a"));
75          method.addParameter(new Parameter(FullyQualifiedJavaType.getIntInstance(), "b"));
76          method.addBodyLine("return a + b + f;");
77          topLevelRecord.addMethod(method);
78  
79          method = new Method("sayIt");
80          method.setReturnType(FullyQualifiedJavaType.getStringInstance());
81          method.addBodyLine("return id + \" \" + description;");
82          topLevelRecord.addMethod(method);
83  
84          InnerClass innerClass = new InnerClass("Fred");
85          innerClass.setVisibility(JavaVisibility.PUBLIC);
86          innerClass.setStatic(true);
87          innerClass.addSuperInterface(new FullyQualifiedJavaType("Flintstone"));
88          topLevelRecord.addInnerClass(innerClass);
89  
90          InnerInterface innerInterface = new InnerInterface("Flintstone");
91          topLevelRecord.addInnerInterface(innerInterface);
92  
93          InnerEnum innerEnum = new InnerEnum("MyEnum");
94          innerEnum.addEnumConstant("A");
95          innerEnum.addEnumConstant("B");
96          innerEnum.addEnumConstant("C");
97          topLevelRecord.addInnerEnum(innerEnum);
98  
99          InnerRecord innerRecord = new InnerRecord("InnerRecord");
100         innerRecord.addParameter(new Parameter(FullyQualifiedJavaType.getIntInstance(), "id"));
101         innerRecord.addParameter(new Parameter(FullyQualifiedJavaType.getStringInstance(), "description"));
102         topLevelRecord.addInnerRecord(innerRecord);
103 
104         return topLevelRecord;
105     }
106 }
