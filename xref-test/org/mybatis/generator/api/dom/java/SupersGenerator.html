16  package org.mybatis.generator.api.dom.java;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  /**
22   * This test verifies that implementations can be in different packages from their super classes.
23   */
24  public class SupersGenerator {
25  
26      private static final String BASE_PACKAGE = "mbg.domtest.generators.supers";
27  
28      public static List<CompilationUnit> generateTestClasses() {
29          List<CompilationUnit> answer = new ArrayList<>();
30  
31          TopLevelClass baseClass = getBaseClass();
32          TopLevelClass superClass = getSuperClass();
33          baseClass.setSuperClass(superClass.getType());
34  
35          Interface baseInterface = getBaseInterface();
36          Interface superInterface = getSuperInterface();
37          baseInterface.addSuperInterface(superInterface.getType());
38          baseClass.addSuperInterface(superInterface.getType());
39  
40          answer.add(baseClass);
41          answer.add(superClass);
42          answer.add(baseInterface);
43          answer.add(superInterface);
44  
45          return answer;
46      }
47  
48      private static TopLevelClass getSuperClass() {
49          FullyQualifiedJavaType fqjt = new FullyQualifiedJavaType(BASE_PACKAGE + ".sub.SuperClass");
50          TopLevelClass tlc = new TopLevelClass(fqjt);
51          tlc.setVisibility(JavaVisibility.PUBLIC);
52  
53          return tlc;
54      }
55  
56      private static Interface getSuperInterface() {
57          FullyQualifiedJavaType fqjt = new FullyQualifiedJavaType(BASE_PACKAGE + ".sub.SuperInterface");
58          Interface ifc = new Interface(fqjt);
59          ifc.setVisibility(JavaVisibility.PUBLIC);
60  
61          return ifc;
62      }
63  
64      private static TopLevelClass getBaseClass() {
65          FullyQualifiedJavaType fqjt = new FullyQualifiedJavaType(BASE_PACKAGE + ".BaseClass");
66          TopLevelClass tlc = new TopLevelClass(fqjt);
67          tlc.setVisibility(JavaVisibility.PUBLIC);
68  
69          return tlc;
70      }
71  
72      private static Interface getBaseInterface() {
73          FullyQualifiedJavaType fqjt = new FullyQualifiedJavaType(BASE_PACKAGE + ".BaseInterface");
74          Interface ifc = new Interface(fqjt);
75          ifc.setVisibility(JavaVisibility.PUBLIC);
76  
77          return ifc;
78      }
79  }
