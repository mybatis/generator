45      @Test
46      void testThatFilesAreTheSameAfterMerge() throws Exception {
47          DefaultXmlFormatter xmlFormatter = new DefaultXmlFormatter();
48          Properties p = new Properties();
49          p.setProperty(PropertyRegistry.COMMENT_GENERATOR_SUPPRESS_DATE, "true");
50          CommentGenerator commentGenerator = new DefaultCommentGenerator();
51          commentGenerator.addConfigurationProperties(p);
52  
53          Document document = new Document(XmlConstants.MYBATIS3_MAPPER_PUBLIC_ID,
54                  XmlConstants.MYBATIS3_MAPPER_SYSTEM_ID);
55          document.setRootElement(getSqlMapElement(commentGenerator));
56  
57          GeneratedXmlFile generatedFile1 = new GeneratedXmlFile(document, "TestMapper.xml", "org.mybatis.test", "src",
58                  true, xmlFormatter);
59          InputSource is1 = new InputSource(new StringReader(generatedFile1.getFormattedContent()));
60  
61          GeneratedXmlFile generatedFile2 = new GeneratedXmlFile(document, "TestMapper.xml", "org.mybatis.test", "src",
62                  true, xmlFormatter);
63          InputSource is2 = new InputSource(new StringReader(generatedFile2.getFormattedContent()));
64  
65          String mergedSource = XmlFileMergerJaxp.getMergedSource(is1, is2, "TestMapper.xml");
66  
67          assertEquals(generatedFile1.getFormattedContent(), mergedSource);
68      }

152     private XmlElement getSqlMapElement(CommentGenerator commentGenerator) {
153 
154         XmlElement answer = new XmlElement("mapper");
155         String namespace = "org.mybatis.test.TestMapper";
156         answer.addAttribute(new Attribute("namespace", namespace));
157 
158         commentGenerator.addRootComment(answer);
159 
160         addInsertElement(commentGenerator, answer);
161         addCdataNode1(commentGenerator, answer);
162         addCdataNode2(commentGenerator, answer);
163 
164         return answer;
165     }
166 
167     private void addInsertElement(CommentGenerator commentGenerator, XmlElement parentElement) {
168         XmlElement answer = new XmlElement("insert");
169 
170         answer.addAttribute(new Attribute("id", "insert"));
171 
172         FullyQualifiedJavaType parameterType;
173         parameterType = new FullyQualifiedJavaType("org.mybatis.test.TestRecord");
174 
175         answer.addAttribute(new Attribute("parameterType", parameterType.getFullyQualifiedName()));
176 
177         commentGenerator.addComment(answer);
178 
179         StringBuilder insertClause = new StringBuilder();
180         StringBuilder valuesClause = new StringBuilder();
181 
182         insertClause.append("insert into ");
183         insertClause.append("myschema.mytable");
184         insertClause.append(" (id, description)");
185 
186         valuesClause.append("values (#{id}, #{description})");
187 
188         answer.addElement(new TextElement(insertClause.toString()));
189 
190         answer.addElement(new TextElement(valuesClause.toString()));
191 
192         parentElement.addElement(answer);
193     }
194 
195     private void addCdataNode1(CommentGenerator commentGenerator, XmlElement parentElement) {
196         XmlElement answer = new XmlElement("select");
197         answer.addAttribute(new Attribute("id", "selectWithCdata1"));
198         commentGenerator.addComment(answer);
199 
200         answer.addElement(new TextElement("<![CDATA["));
201         answer.addElement(new TextElement("select foo from bar where foo < 22"));
202         answer.addElement(new TextElement("]]>"));
203 
204         parentElement.addElement(answer);
205     }
206 
207     private void addCdataNode2(CommentGenerator commentGenerator, XmlElement parentElement) {
208         XmlElement answer = new XmlElement("select");
209         answer.addAttribute(new Attribute("id", "selectWithCdata2"));
210         commentGenerator.addComment(answer);
211 
212         answer.addElement(new TextElement("select foo from bar where foo <![CDATA[ < ]]> 22"));
213 
214         parentElement.addElement(answer);
215     }
