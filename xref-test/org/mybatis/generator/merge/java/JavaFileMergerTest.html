16  package org.mybatis.generator.merge.java;
17  
18  import static org.assertj.core.api.Assertions.assertThat;
19  import static org.assertj.core.api.Assertions.assertThatExceptionOfType;
20  import static org.mybatis.generator.internal.util.messages.Messages.getString;
21  
22  import java.util.stream.Stream;
23  
24  import org.assertj.core.api.InstanceOfAssertFactories;
25  import org.junit.jupiter.api.Test;
26  import org.junit.jupiter.params.ParameterizedTest;
27  import org.junit.jupiter.params.provider.Arguments;
28  import org.junit.jupiter.params.provider.MethodSource;
29  import org.mybatis.generator.exception.MergeException;
30  import org.mybatis.generator.merge.MergeTestCase;
31  
32  class JavaFileMergerTest {
33      @ParameterizedTest
34      @MethodSource("mergeTestCases")
35      void mergeTestCases(JavaMergeTestCase testCase, String parameter) throws Exception {
36          JavaFileMerger javaFileMerger = JavaMergerFactory.getMerger(testCase.printerConfiguration());
37          var actual = javaFileMerger.getMergedSource(testCase.newContent(parameter),
38                  testCase.existingContent(parameter));
39          assertThat(actual).isEqualToNormalizingNewlines(testCase.expectedContentAfterMerge(parameter));
40      }
41  
42      static Stream<Arguments> mergeTestCases() {
43          return MergeTestCase.findTestCases("org.mybatis.generator.merge.java");
44      }
45  
46      @Test
47      void testBadExistingFile() {
48          JavaFileMerger javaFileMerger = JavaMergerFactory.getMerger(JavaMergerFactory.PrinterConfiguration.ECLIPSE);
49          String badExistingFile = "some random text";
50  
51          assertThatExceptionOfType(MergeException.class).isThrownBy(() ->
52                  javaFileMerger.getMergedSource(badExistingFile, badExistingFile))
53                  .withMessage(getString("RuntimeError.28", "existing Java file"))
54                  .extracting(MergeException::getExtraMessages).asInstanceOf(InstanceOfAssertFactories.LIST)
55                  .hasSize(1);
56      }
57  
58      @Test
59      void testBadNewFile() {
60          JavaFileMerger javaFileMerger = JavaMergerFactory.getMerger(JavaMergerFactory.PrinterConfiguration.ECLIPSE);
61          String existingFile = "public class Foo { public int i; }";
62          String badNewFile = "some random text";
63  
64          assertThatExceptionOfType(MergeException.class).isThrownBy(() ->
65                          javaFileMerger.getMergedSource(badNewFile, existingFile))
66                  .withMessage(getString("RuntimeError.28", "new Java file"))
67                  .extracting(MergeException::getExtraMessages).asInstanceOf(InstanceOfAssertFactories.LIST)
68                  .hasSize(1);
69      }
70  
71      @Test
72      void testNoTypeInExistingFile() {
73          JavaFileMerger javaFileMerger = JavaMergerFactory.getMerger(JavaMergerFactory.PrinterConfiguration.ECLIPSE);
74          String existingFileNoTypes = "package foo.bar;";
75  
76          assertThatExceptionOfType(MergeException.class).isThrownBy(() ->
77                          javaFileMerger.getMergedSource(existingFileNoTypes, existingFileNoTypes))
78                  .withMessage(getString("RuntimeError.29", "existing Java file"));
79      }
80  }
