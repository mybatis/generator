46      /**
47       * This object is used to hold information related to the table itself, not the columns in the
48       * table.
49       *
50       * @param introspectedCatalog
51       *            the actual catalog of the table as returned from DatabaseMetaData. This value
52       *            should only be set if the user configured a catalog. Otherwise the
53       *            DatabaseMetaData is reporting some database default that we don't want in the
54       *            generated code.
55       * @param introspectedSchema
56       *            the actual schema of the table as returned from DatabaseMetaData. This value
57       *            should only be set if the user configured a schema. Otherwise the
58       *            DatabaseMetaData is reporting some database default that we don't want in the
59       *            generated code.
60       * @param introspectedTableName
61       *            the actual table name as returned from DatabaseMetaData
62       * @param domainObjectName
63       *            the configured domain object name for this table. If nothing is configured, we'll build the domain
64       *            object named based on the tableName or runtimeTableName.
65       * @param alias
66       *            a configured alias for the table. This alias will be added to the table name in the SQL
67       * @param ignoreQualifiersAtRuntime
68       *            if true, then the catalog and schema qualifiers will be ignored when composing fully qualified names
69       *            in the generated SQL. This is used, for example, when the user needs to specify a specific schema for
70       *            generating code but does not want the schema in the generated SQL
71       * @param runtimeCatalog
72       *            this is used to "rename" the catalog in the generated SQL. This is useful, for example, when
73       *            generating code against one catalog that should run with a different catalog.
74       * @param runtimeSchema
75       *            this is used to "rename" the schema in the generated SQL. This is useful, for example, when generating
76       *            code against one schema that should run with a different schema.
77       * @param runtimeTableName
78       *            this is used to "rename" the table in the generated SQL. This is useful, for example, when generating
79       *            code to run with an Oracle synonym. The user would have to specify the actual table name and schema
80       *            for generation, but would want to use the synonym name in the generated SQL
81       * @param delimitIdentifiers
82       *            if true, then the table identifiers will be delimited at runtime. The delimiter characters are
83       *            obtained from the Context.
84       * @param domainObjectRenamingRule
85       *            If domainObjectName is not configured, we'll build the domain object named based on the tableName
86       *            or runtimeTableName.
87       *            And then we use the domain object renaming rule to generate the final domain object name.
88       * @param context
89       *            the context
90       */
91      public FullyQualifiedTable(String introspectedCatalog,
92              String introspectedSchema, String introspectedTableName,
93              String domainObjectName, String alias,
94              boolean ignoreQualifiersAtRuntime, String runtimeCatalog,
95              String runtimeSchema, String runtimeTableName,
96              boolean delimitIdentifiers, DomainObjectRenamingRule domainObjectRenamingRule,
97              Context context) {
98          super();
99          this.introspectedCatalog = introspectedCatalog;
100         this.introspectedSchema = introspectedSchema;
101         this.introspectedTableName = introspectedTableName;
102         this.ignoreQualifiersAtRuntime = ignoreQualifiersAtRuntime;
103         this.runtimeCatalog = runtimeCatalog;
104         this.runtimeSchema = runtimeSchema;
105         this.runtimeTableName = runtimeTableName;
106         this.domainObjectRenamingRule = domainObjectRenamingRule;
107 
108         if (stringHasValue(domainObjectName)) {
109             int index = domainObjectName.lastIndexOf('.');
110             if (index == -1) {
111                 this.domainObjectName = domainObjectName;
112             } else {
113                 this.domainObjectName = domainObjectName.substring(index + 1);
114                 this.domainObjectSubPackage = domainObjectName.substring(0, index);
115             }
116         }
117 
118         if (alias == null) {
119             this.alias = null;
120         } else {
121             this.alias = alias.trim();
122         }
123 
124         beginningDelimiter = delimitIdentifiers ? context
125                 .getBeginningDelimiter() : ""; //$NON-NLS-1$
126         endingDelimiter = delimitIdentifiers ? context.getEndingDelimiter()
127                 : ""; //$NON-NLS-1$
128     }

142     public String getFullyQualifiedTableNameAtRuntime() {
143         StringBuilder localCatalog = new StringBuilder();
144         if (!ignoreQualifiersAtRuntime) {
145             if (stringHasValue(runtimeCatalog)) {
146                 localCatalog.append(runtimeCatalog);
147             } else if (stringHasValue(introspectedCatalog)) {
148                 localCatalog.append(introspectedCatalog);
149             }
150         }
151         if (localCatalog.length() > 0) {
152             addDelimiters(localCatalog);
153         }
154 
155         StringBuilder localSchema = new StringBuilder();
156         if (!ignoreQualifiersAtRuntime) {
157             if (stringHasValue(runtimeSchema)) {
158                 localSchema.append(runtimeSchema);
159             } else if (stringHasValue(introspectedSchema)) {
160                 localSchema.append(introspectedSchema);
161             }
162         }
163         if (localSchema.length() > 0) {
164             addDelimiters(localSchema);
165         }
166 
167         StringBuilder localTableName = new StringBuilder();
168         if (stringHasValue(runtimeTableName)) {
169             localTableName.append(runtimeTableName);
170         } else {
171             localTableName.append(introspectedTableName);
172         }
173         addDelimiters(localTableName);
174 
175         return composeFullyQualifiedTableName(localCatalog
176                 .toString(), localSchema.toString(), localTableName.toString(),
177                 '.');
178     }

193     public String getDomainObjectName() {
194         if (stringHasValue(domainObjectName)) {
195             return domainObjectName;
196         }
197 
198         String finalDomainObjectName;
199         if (stringHasValue(runtimeTableName)) {
200             finalDomainObjectName = JavaBeansUtil.getCamelCaseString(runtimeTableName, true);
201         } else {
202             finalDomainObjectName = JavaBeansUtil.getCamelCaseString(introspectedTableName, true);
203         }
204 
205         if (domainObjectRenamingRule != null) {
206             Pattern pattern = Pattern.compile(domainObjectRenamingRule.getSearchString());
207             String replaceString = domainObjectRenamingRule.getReplaceString();
208             replaceString = replaceString == null ? "" : replaceString; //$NON-NLS-1$
209             Matcher matcher = pattern.matcher(finalDomainObjectName);
210             finalDomainObjectName = JavaBeansUtil.getFirstCharacterUppercase(matcher.replaceAll(replaceString));
211         }
212         return finalDomainObjectName;
213     }

253     /**
254      * Calculates a Java package fragment based on the table catalog and schema.
255      * If qualifiers are ignored, then this method will return an empty string.
256      * 
257      * <p>This method is used for determining the sub package for Java client and
258      * SQL map (XML) objects.  It ignores any sub-package added to the
259      * domain object name in the table configuration.
260      *
261      * @param isSubPackagesEnabled
262      *            the is sub packages enabled
263      * @return the subpackage for this table
264      */
265     public String getSubPackageForClientOrSqlMap(boolean isSubPackagesEnabled) {
266         StringBuilder sb = new StringBuilder();
267         if (!ignoreQualifiersAtRuntime && isSubPackagesEnabled) {
268             if (stringHasValue(runtimeCatalog)) {
269                 sb.append('.');
270                 sb.append(runtimeCatalog.toLowerCase());
271             } else if (stringHasValue(introspectedCatalog)) {
272                 sb.append('.');
273                 sb.append(introspectedCatalog.toLowerCase());
274             }
275 
276             if (stringHasValue(runtimeSchema)) {
277                 sb.append('.');
278                 sb.append(runtimeSchema.toLowerCase());
279             } else if (stringHasValue(introspectedSchema)) {
280                 sb.append('.');
281                 sb.append(introspectedSchema.toLowerCase());
282             }
283         }
284 
285         // TODO - strip characters that are not valid in package names
286         return sb.toString();
287     }

289     /**
290      * Calculates a Java package fragment based on the table catalog and schema.
291      * If qualifiers are ignored, then this method will return an empty string.
292      * 
293      * <p>This method is used for determining the sub package for Java model objects only.
294      * It takes into account the possibility that a sub-package was added to the
295      * domain object name in the table configuration.
296      *
297      * @param isSubPackagesEnabled
298      *            the is sub packages enabled
299      * @return the subpackage for this table
300      */
301     public String getSubPackageForModel(boolean isSubPackagesEnabled) {
302         StringBuilder sb = new StringBuilder();
303         sb.append(getSubPackageForClientOrSqlMap(isSubPackagesEnabled));
304 
305         if (stringHasValue(domainObjectSubPackage)) {
306             sb.append('.');
307             sb.append(domainObjectSubPackage);
308         }
309 
310         return sb.toString();
311     }

313     private void addDelimiters(StringBuilder sb) {
314         if (stringHasValue(beginningDelimiter)) {
315             sb.insert(0, beginningDelimiter);
316         }
317 
318         if (stringHasValue(endingDelimiter)) {
319             sb.append(endingDelimiter);
320         }
321     }
