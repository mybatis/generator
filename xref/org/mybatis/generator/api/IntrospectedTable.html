145     public IntrospectedColumn getColumn(String columnName) {
146         if (columnName == null) {
147             return null;
148         } else {
149             // search primary key columns
150             for (IntrospectedColumn introspectedColumn : primaryKeyColumns) {
151                 if (introspectedColumn.isColumnNameDelimited()) {
152                     if (introspectedColumn.getActualColumnName().equals(
153                             columnName)) {
154                         return introspectedColumn;
155                     }
156                 } else {
157                     if (introspectedColumn.getActualColumnName()
158                             .equalsIgnoreCase(columnName)) {
159                         return introspectedColumn;
160                     }
161                 }
162             }
163 
164             // search base columns
165             for (IntrospectedColumn introspectedColumn : baseColumns) {
166                 if (introspectedColumn.isColumnNameDelimited()) {
167                     if (introspectedColumn.getActualColumnName().equals(
168                             columnName)) {
169                         return introspectedColumn;
170                     }
171                 } else {
172                     if (introspectedColumn.getActualColumnName()
173                             .equalsIgnoreCase(columnName)) {
174                         return introspectedColumn;
175                     }
176                 }
177             }
178 
179             // search blob columns
180             for (IntrospectedColumn introspectedColumn : blobColumns) {
181                 if (introspectedColumn.isColumnNameDelimited()) {
182                     if (introspectedColumn.getActualColumnName().equals(
183                             columnName)) {
184                         return introspectedColumn;
185                     }
186                 } else {
187                     if (introspectedColumn.getActualColumnName()
188                             .equalsIgnoreCase(columnName)) {
189                         return introspectedColumn;
190                     }
191                 }
192             }
193 
194             return null;
195         }
196     }

199      * Returns true if any of the columns in the table are JDBC Dates (as
200      * opposed to timestamps).
201      * 
202      * @return true if the table contains DATE columns
203      */
204     public boolean hasJDBCDateColumns() {
205         boolean rc = false;
206 
207         for (IntrospectedColumn introspectedColumn : primaryKeyColumns) {
208             if (introspectedColumn.isJDBCDateColumn()) {
209                 rc = true;
210                 break;
211             }
212         }
213 
214         if (!rc) {
215             for (IntrospectedColumn introspectedColumn : baseColumns) {
216                 if (introspectedColumn.isJDBCDateColumn()) {
217                     rc = true;
218                     break;
219                 }
220             }
221         }
222 
223         return rc;
224     }

446     public void addPrimaryKeyColumn(String columnName) {
447         boolean found = false;
448         // first search base columns
449         Iterator<IntrospectedColumn> iter = baseColumns.iterator();
450         while (iter.hasNext()) {
451             IntrospectedColumn introspectedColumn = iter.next();
452             if (introspectedColumn.getActualColumnName().equals(columnName)) {
453                 primaryKeyColumns.add(introspectedColumn);
454                 iter.remove();
455                 found = true;
456                 break;
457             }
458         }
459 
460         // search blob columns in the weird event that a blob is the primary key
461         if (!found) {
462             iter = blobColumns.iterator();
463             while (iter.hasNext()) {
464                 IntrospectedColumn introspectedColumn = iter.next();
465                 if (introspectedColumn.getActualColumnName().equals(columnName)) {
466                     primaryKeyColumns.add(introspectedColumn);
467                     iter.remove();
468                     found = true;
469                     break;
470                 }
471             }
472         }
473     }
