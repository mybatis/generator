85      /**
86       * Returns the fully qualified name - including any generic type parameters.
87       *
88       * @return Returns the fullyQualifiedName.
89       */
90      public String getFullyQualifiedName() {
91          StringBuilder sb = new StringBuilder();
92          if (wildcardType) {
93              sb.append('?');
94              if (boundedWildcard) {
95                  if (extendsBoundedWildcard) {
96                      sb.append(" extends "); //$NON-NLS-1$
97                  } else {
98                      sb.append(" super "); //$NON-NLS-1$
99                  }
100 
101                 sb.append(baseQualifiedName);
102             }
103         } else {
104             sb.append(baseQualifiedName);
105         }
106 
107         if (!typeArguments.isEmpty()) {
108             boolean first = true;
109             sb.append('<');
110             for (FullyQualifiedJavaType fqjt : typeArguments) {
111                 if (first) {
112                     first = false;
113                 } else {
114                     sb.append(", "); //$NON-NLS-1$
115                 }
116                 sb.append(fqjt.getFullyQualifiedName());
117 
118             }
119             sb.append('>');
120         }
121 
122         return sb.toString();
123     }

129     /**
130      * Returns a list of Strings that are the fully qualified names of this type, and any generic type argument
131      * associated with this type.
132      *
133      * @return the import list
134      */
135     public List<String> getImportList() {
136         List<String> answer = new ArrayList<>();
137         if (isExplicitlyImported()) {
138             int index = baseShortName.indexOf('.');
139             if (index == -1) {
140                 answer.add(calculateActualImport(baseQualifiedName));
141             } else {
142                 // an inner class is specified, only import the top
143                 // level class
144                 String sb = packageName + '.' + calculateActualImport(baseShortName.substring(0, index));
145                 answer.add(sb);
146             }
147         }
148 
149         for (FullyQualifiedJavaType fqjt : typeArguments) {
150             answer.addAll(fqjt.getImportList());
151         }
152 
153         return answer;
154     }

156     private String calculateActualImport(String name) {
157         String answer = name;
158         if (this.isArray()) {
159             int index = name.indexOf('[');
160             if (index != -1) {
161                 answer = name.substring(0, index);
162             }
163         }
164         return answer;
165     }

329     private void parse(String fullTypeSpecification) {
330         String spec = fullTypeSpecification.trim();
331 
332         if (spec.startsWith("?")) { //$NON-NLS-1$
333             wildcardType = true;
334             spec = spec.substring(1).trim();
335             if (spec.startsWith("extends ")) { //$NON-NLS-1$
336                 boundedWildcard = true;
337                 extendsBoundedWildcard = true;
338                 spec = spec.substring(8);  // "extends ".length()
339             } else if (spec.startsWith("super ")) { //$NON-NLS-1$
340                 boundedWildcard = true;
341                 extendsBoundedWildcard = false;
342                 spec = spec.substring(6);  // "super ".length()
343             } else {
344                 boundedWildcard = false;
345             }
346             parse(spec);
347         } else {
348             int index = fullTypeSpecification.indexOf('<');
349             if (index == -1) {
350                 simpleParse(fullTypeSpecification);
351             } else {
352                 simpleParse(fullTypeSpecification.substring(0, index));
353                 int endIndex = fullTypeSpecification.lastIndexOf('>');
354                 if (endIndex == -1) {
355                     throw new RuntimeException(getString(
356                             "RuntimeError.22", fullTypeSpecification)); //$NON-NLS-1$
357                 }
358                 genericParse(fullTypeSpecification.substring(index, endIndex + 1));
359             }
360 
361             // this is far from a perfect test for detecting arrays, but is close
362             // enough for most cases.  It will not detect an improperly specified
363             // array type like byte], but it will detect byte[] and byte[   ]
364             // which are both valid
365             isArray = fullTypeSpecification.endsWith("]"); //$NON-NLS-1$
366         }
367     }

369     private void simpleParse(String typeSpecification) {
370         baseQualifiedName = typeSpecification.trim();
371         if (baseQualifiedName.contains(".")) { //$NON-NLS-1$
372             packageName = getPackage(baseQualifiedName);
373             baseShortName = baseQualifiedName
374                     .substring(packageName.length() + 1);
375             int index = baseShortName.lastIndexOf('.');
376             if (index != -1) {
377                 baseShortName = baseShortName.substring(index + 1);
378             }
379 
380             //$NON-NLS-1$
381             explicitlyImported = !JAVA_LANG.equals(packageName);
382         } else {
383             baseShortName = baseQualifiedName;
384             explicitlyImported = false;
385             packageName = ""; //$NON-NLS-1$
386 
387             switch (baseQualifiedName) {
388             case "byte":  //$NON-NLS-1$
389                 primitive = true;
390                 primitiveTypeWrapper = PrimitiveTypeWrapper.getByteInstance();
391                 break;
392             case "short":  //$NON-NLS-1$
393                 primitive = true;
394                 primitiveTypeWrapper = PrimitiveTypeWrapper.getShortInstance();
395                 break;
396             case "int":  //$NON-NLS-1$
397                 primitive = true;
398                 primitiveTypeWrapper = PrimitiveTypeWrapper.getIntegerInstance();
399                 break;
400             case "long":  //$NON-NLS-1$
401                 primitive = true;
402                 primitiveTypeWrapper = PrimitiveTypeWrapper.getLongInstance();
403                 break;
404             case "char":  //$NON-NLS-1$
405                 primitive = true;
406                 primitiveTypeWrapper = PrimitiveTypeWrapper.getCharacterInstance();
407                 break;
408             case "float":  //$NON-NLS-1$
409                 primitive = true;
410                 primitiveTypeWrapper = PrimitiveTypeWrapper.getFloatInstance();
411                 break;
412             case "double":  //$NON-NLS-1$
413                 primitive = true;
414                 primitiveTypeWrapper = PrimitiveTypeWrapper.getDoubleInstance();
415                 break;
416             case "boolean":  //$NON-NLS-1$
417                 primitive = true;
418                 primitiveTypeWrapper = PrimitiveTypeWrapper.getBooleanInstance();
419                 break;
420             default:
421                 primitive = false;
422                 primitiveTypeWrapper = null;
423                 break;
424             }
425         }
426     }

428     private void genericParse(String genericSpecification) {
429         int lastIndex = genericSpecification.lastIndexOf('>');
430         if (lastIndex == -1) {
431             // shouldn't happen - should be caught already, but just in case...
432             throw new RuntimeException(getString(
433                     "RuntimeError.22", genericSpecification)); //$NON-NLS-1$
434         }
435         String argumentString = genericSpecification.substring(1, lastIndex);
436         // need to find "," outside of a <> bounds
437         StringTokenizer st = new StringTokenizer(argumentString, ",<>", true); //$NON-NLS-1$
438         int openCount = 0;
439         StringBuilder sb = new StringBuilder();
440         while (st.hasMoreTokens()) {
441             String token = st.nextToken();
442             if ("<".equals(token)) { //$NON-NLS-1$
443                 sb.append(token);
444                 openCount++;
445             } else if (">".equals(token)) { //$NON-NLS-1$
446                 sb.append(token);
447                 openCount--;
448             } else if (",".equals(token)) { //$NON-NLS-1$
449                 if (openCount == 0) {
450                     typeArguments
451                             .add(new FullyQualifiedJavaType(sb.toString()));
452                     sb.setLength(0);
453                 } else {
454                     sb.append(token);
455                 }
456             } else {
457                 sb.append(token);
458             }
459         }
460 
461         if (openCount != 0) {
462             throw new RuntimeException(getString(
463                     "RuntimeError.22", genericSpecification)); //$NON-NLS-1$
464         }
465 
466         String finalType = sb.toString();
467         if (stringHasValue(finalType)) {
468             typeArguments.add(new FullyQualifiedJavaType(finalType));
469         }
470     }

472     /**
473      * Returns the package name of a fully qualified type.
474      *
475      * <p>This method calculates the package as the part of the fully qualified name up to, but not including, the last
476      * element. Therefore, it does not support fully qualified inner classes. Not totally fool proof, but correct in
477      * most instances.
478      *
479      * @param baseQualifiedName
480      *            the base qualified name
481      * @return the package
482      */
483     private static String getPackage(String baseQualifiedName) {
484         int index = baseQualifiedName.lastIndexOf('.');
485         return baseQualifiedName.substring(0, index);
486     }
