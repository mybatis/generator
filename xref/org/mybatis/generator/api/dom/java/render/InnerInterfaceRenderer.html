16  package org.mybatis.generator.api.dom.java.render;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  import org.mybatis.generator.api.dom.java.CompilationUnit;
22  import org.mybatis.generator.api.dom.java.InnerInterface;
23  import org.mybatis.generator.api.dom.java.JavaDomUtils;
24  import org.mybatis.generator.internal.util.CustomCollectors;
25  
26  public class InnerInterfaceRenderer {
27  
28      public List<String> render(InnerInterface innerInterface, CompilationUnit compilationUnit) {
29          List<String> lines = new ArrayList<>();
30  
31          lines.addAll(innerInterface.getJavaDocLines());
32          lines.addAll(innerInterface.getAnnotations());
33          lines.add(renderFirstLine(innerInterface, compilationUnit));
34          lines.addAll(RenderingUtilities.renderFields(innerInterface.getFields(), compilationUnit));
35          lines.addAll(RenderingUtilities.renderInterfaceMethods(innerInterface.getMethods(), compilationUnit));
36          lines.addAll(RenderingUtilities.renderInnerClasses(innerInterface.getInnerClasses(), compilationUnit));
37          lines.addAll(RenderingUtilities.renderInnerInterfaces(innerInterface.getInnerInterfaces(), compilationUnit));
38          lines.addAll(RenderingUtilities.renderInnerEnums(innerInterface.getInnerEnums(), compilationUnit));
39  
40          lines = RenderingUtilities.removeLastEmptyLine(lines);
41  
42          lines.add("}"); //$NON-NLS-1$
43  
44          return lines;
45      }
46  
47      private String renderFirstLine(InnerInterface innerInterface, CompilationUnit compilationUnit) {
48          StringBuilder sb = new StringBuilder();
49  
50          sb.append(innerInterface.getVisibility().getValue());
51  
52          if (innerInterface.isStatic()) {
53              sb.append("static "); //$NON-NLS-1$
54          }
55  
56          sb.append("interface "); //$NON-NLS-1$
57          sb.append(innerInterface.getType().getShortName());
58          sb.append(RenderingUtilities.renderTypeParameters(innerInterface.getTypeParameters(), compilationUnit));
59          sb.append(renderSuperInterfaces(innerInterface, compilationUnit));
60          sb.append(" {"); //$NON-NLS-1$
61  
62          return sb.toString();
63      }
64  
65      // should return an empty string if no super interfaces
66      private String renderSuperInterfaces(InnerInterface innerInterface, CompilationUnit compilationUnit) {
67          return innerInterface.getSuperInterfaceTypes().stream()
68                  .map(tp -> JavaDomUtils.calculateTypeName(compilationUnit, tp))
69                  .collect(CustomCollectors.joining(", ", " extends ", "")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
70      }
71  }
