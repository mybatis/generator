29  import org.mybatis.generator.api.dom.java.CompilationUnit;

40  public class RenderingUtilities {
41      private RenderingUtilities() {}
42  
43      public static final String JAVA_INDENT = "    "; //$NON-NLS-1$
44      private static final TypeParameterRenderer typeParameterRenderer = new TypeParameterRenderer();
45      private static final FieldRenderer fieldRenderer = new FieldRenderer();
46      private static final InitializationBlockRenderer initializationBlockRenderer = new InitializationBlockRenderer();
47      private static final MethodRenderer methodRenderer = new MethodRenderer();
48      private static final InnerClassRenderer innerClassRenderer = new InnerClassRenderer();
49      private static final InnerInterfaceRenderer innerInterfaceRenderer = new InnerInterfaceRenderer();
50      private static final InnerEnumRenderer innerEnumRenderer = new InnerEnumRenderer();
51  
52      // should return an empty string if no type parameters
53      public static String renderTypeParameters(List<TypeParameter> typeParameters, CompilationUnit compilationUnit) {
54          return typeParameters.stream()
55                  .map(tp -> typeParameterRenderer.render(tp, compilationUnit))
56                  .collect(CustomCollectors.joining(", ", "<", "> ")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
57      }
58  
59      public static List<String> renderFields(List<Field> fields, CompilationUnit compilationUnit) {
60          return fields.stream()
61                  .flatMap(f -> renderField(f, compilationUnit))
62                  .collect(Collectors.toList());
63      }
64  
65      private static Stream<String> renderField(Field field, CompilationUnit compilationUnit) {
66          return addEmptyLine(fieldRenderer.render(field, compilationUnit).stream()
67                  .map(RenderingUtilities::javaIndent));
68      }
69  
70      public static List<String> renderInitializationBlocks(List<InitializationBlock> initializationBlocks) {
71          return initializationBlocks.stream()
72                  .flatMap(RenderingUtilities::renderInitializationBlock)
73                  .collect(Collectors.toList());
74      }
75  
76      private static Stream<String> renderInitializationBlock(InitializationBlock initializationBlock) {
77          return addEmptyLine(initializationBlockRenderer.render(initializationBlock).stream()
78                  .map(RenderingUtilities::javaIndent));
79      }
80  
81      public static List<String> renderClassOrEnumMethods(List<Method> methods, CompilationUnit compilationUnit) {
82          return methods.stream()
83                  .flatMap(m -> renderMethod(m, false, compilationUnit))
84                  .collect(Collectors.toList());
85      }
86  
87      public static List<String> renderInterfaceMethods(List<Method> methods, CompilationUnit compilationUnit) {
88          return methods.stream()
89                  .flatMap(m -> renderMethod(m, true, compilationUnit))
90                  .collect(Collectors.toList());
91      }
92  
93      private static Stream<String> renderMethod(Method method, boolean inInterface, CompilationUnit compilationUnit) {
94          return addEmptyLine(methodRenderer.render(method, inInterface, compilationUnit).stream()
95                  .map(RenderingUtilities::javaIndent));
96      }
97  
98      private static Stream<String> addEmptyLine(Stream<String> in) {
99          return Stream.of(in, Stream.of("")) //$NON-NLS-1$
100                 .flatMap(Function.identity());
101     }
102 
103     public static List<String> renderInnerClasses(List<InnerClass> innerClasses, CompilationUnit compilationUnit) {
104         return innerClasses.stream()
105                 .flatMap(ic -> renderInnerClass(ic, compilationUnit))
106                 .collect(Collectors.toList());
107     }
108 
109     public static List<String> renderInnerClassNoIndent(InnerClass innerClass, CompilationUnit compilationUnit) {
110         return innerClassRenderer.render(innerClass, compilationUnit);
111     }
112 
113     private static Stream<String> renderInnerClass(InnerClass innerClass, CompilationUnit compilationUnit) {
114         return addEmptyLine(innerClassRenderer.render(innerClass, compilationUnit).stream()
115                 .map(RenderingUtilities::javaIndent));
116     }
117 
118     public static List<String> renderInnerInterfaces(List<InnerInterface> innerInterfaces,
119             CompilationUnit compilationUnit) {
120         return innerInterfaces.stream()
121                 .flatMap(ii -> renderInnerInterface(ii, compilationUnit))
122                 .collect(Collectors.toList());
123     }
124 
125     public static List<String> renderInnerInterfaceNoIndent(InnerInterface innerInterface,
126             CompilationUnit compilationUnit) {
127         return innerInterfaceRenderer.render(innerInterface, compilationUnit);
128     }
129 
130     private static Stream<String> renderInnerInterface(InnerInterface innerInterface, CompilationUnit compilationUnit) {
131         return addEmptyLine(innerInterfaceRenderer.render(innerInterface, compilationUnit).stream()
132                 .map(RenderingUtilities::javaIndent));
133     }
134 
135     public static List<String> renderInnerEnums(List<InnerEnum> innerEnums, CompilationUnit compilationUnit) {
136         return innerEnums.stream()
137                 .flatMap(ie -> renderInnerEnum(ie, compilationUnit))
138                 .collect(Collectors.toList());
139     }
140 
141     public static List<String> renderInnerEnumNoIndent(InnerEnum innerEnum, CompilationUnit compilationUnit) {
142         return innerEnumRenderer.render(innerEnum, compilationUnit);
143     }
144 
145     private static Stream<String> renderInnerEnum(InnerEnum innerEnum, CompilationUnit compilationUnit) {
146         return addEmptyLine(innerEnumRenderer.render(innerEnum, compilationUnit).stream()
147                 .map(RenderingUtilities::javaIndent));
148     }
149 
150     public static List<String> renderPackage(CompilationUnit compilationUnit) {
151         List<String> answer = new ArrayList<>();
152 
153         String pack = compilationUnit.getType().getPackageName();
154         if (stringHasValue(pack)) {
155             answer.add("package " + pack + ";"); //$NON-NLS-1$ //$NON-NLS-2$
156             answer.add(""); //$NON-NLS-1$
157         }
158         return answer;
159     }
160 
161     public static List<String> renderStaticImports(CompilationUnit compilationUnit) {
162         if (compilationUnit.getStaticImports().isEmpty()) {
163             return Collections.emptyList();
164         }
165 
166         return addEmptyLine(compilationUnit.getStaticImports().stream()
167                 .map(s -> "import static " + s + ";")) //$NON-NLS-1$ //$NON-NLS-2$
168                 .collect(Collectors.toList());
169     }
170 
171     public static List<String> renderImports(CompilationUnit compilationUnit) {
172         Set<String> imports = renderImports(compilationUnit.getImportedTypes());
173 
174         if (imports.isEmpty()) {
175             return Collections.emptyList();
176         }
177 
178         return addEmptyLine(imports.stream()).collect(Collectors.toList());
179     }
180 
181     private static Set<String> renderImports(Set<FullyQualifiedJavaType> imports) {
182         return imports.stream()
183                 .map(FullyQualifiedJavaType::getImportList)
184                 .flatMap(List::stream)
185                 .map(RenderingUtilities::toFullImport)
186                 .collect(Collectors.toCollection(TreeSet::new));
187     }
188 
189     private static String toFullImport(String s) {
190         return "import " + s + ";"; //$NON-NLS-1$ //$NON-NLS-2$
191     }
192 
193 
194     private static String javaIndent(String in) {
195         if (in.isEmpty()) {
196             return in; // don't indent empty lines
197         }
198 
199         return JAVA_INDENT + in;
200     }
201 
202     public static List<String> removeLastEmptyLine(List<String> lines) {
203         if (lines.get(lines.size() - 1).isEmpty()) {
204             return lines.subList(0, lines.size() - 1);
205         } else {
206             return lines;
207         }
208     }
