20  import java.util.List;

23  import org.jspecify.annotations.Nullable;
24  import org.mybatis.generator.api.CommentGenerator;
25  import org.mybatis.generator.api.JavaFormatter;
26  import org.mybatis.generator.api.KnownRuntime;
27  import org.mybatis.generator.api.KotlinFormatter;
28  import org.mybatis.generator.api.Plugin;
29  import org.mybatis.generator.api.XmlFormatter;

31  import org.mybatis.generator.config.Defaults;
32  import org.mybatis.generator.config.PropertyRegistry;
33  import org.mybatis.generator.internal.ObjectFactory;
34  import org.mybatis.generator.internal.PluginAggregator;
35  
36  /**
37   * This class holds common objects in a context that can be pre-calculated before code generation runs.
38   */
39  public class CalculatedContextValues {
40      private final Context context;
41      private final JavaFormatter javaFormatter;
42      private final KotlinFormatter kotlinFormatter;
43      private final XmlFormatter xmlFormatter;
44      private final CommentGenerator commentGenerator;
45      private final @Nullable String javaFileEncoding;
46      private final @Nullable String kotlinFileEncoding;
47      private final PluginAggregator pluginAggregator;
48      private final KnownRuntime knownRuntime;
49      private final String runtimeBuilderClassName;
50  
51      protected CalculatedContextValues(Builder builder) {
52          context = Objects.requireNonNull(builder.context);
53          javaFormatter = ObjectFactory.createJavaFormatter(context);
54          kotlinFormatter = ObjectFactory.createKotlinFormatter(context);
55          xmlFormatter = ObjectFactory.createXmlFormatter(context);
56          commentGenerator = ObjectFactory.createCommentGenerator(context);
57          javaFileEncoding = context.getProperty(PropertyRegistry.CONTEXT_JAVA_FILE_ENCODING);
58          kotlinFileEncoding = context.getProperty(PropertyRegistry.CONTEXT_KOTLIN_FILE_ENCODING);
59          Objects.requireNonNull(builder.warnings);
60  
61          // this will either be the alias entered in the configuration, or the default if nothing was specified
62          String builderAlias = context.getTargetRuntime().orElse(Defaults.DEFAULT_RUNTIME.getAlias());
63          // this will either be a successful lookup by alias, or UNKNOWN
64          knownRuntime = KnownRuntime.getByAlias(builderAlias);
65  
66          pluginAggregator = new PluginAggregator();
67          context.pluginConfigurations().forEach(pluginConfiguration -> {
68              Plugin plugin = ObjectFactory.createPlugin(context, pluginConfiguration, commentGenerator, knownRuntime);
69              if (plugin.validate(builder.warnings)) {
70                  pluginAggregator.addPlugin(plugin);
71              } else {
72                  builder.warnings.add(getString("Warning.24", //$NON-NLS-1$
73                          pluginConfiguration.getConfigurationType()
74                                  .orElse("Unknown Plugin Type"), context.getId())); //$NON-NLS-1$
75              }
76          });
77  
78          if (knownRuntime == KnownRuntime.UNKNOWN) {
79              runtimeBuilderClassName = builderAlias;
80          } else {
81              runtimeBuilderClassName = knownRuntime.getBuilderClassName();
82          }
83      }
84  
85      public Context context() {
86          return context;
87      }
88  
89      public JavaFormatter javaFormatter() {
90          return javaFormatter;
91      }
92  
93      public KotlinFormatter kotlinFormatter() {
94          return kotlinFormatter;
95      }
96  
97      public XmlFormatter xmlFormatter() {
98          return xmlFormatter;
99      }
100 
101     public CommentGenerator commentGenerator() {
102         return commentGenerator;
103     }
104 
105     public @Nullable String javaFileEncoding() {
106         return javaFileEncoding;
107     }
108 
109     public @Nullable String kotlinFileEncoding() {
110         return kotlinFileEncoding;
111     }
112 
113     public PluginAggregator pluginAggregator() {
114         return pluginAggregator;
115     }
116 
117     public KnownRuntime knownRuntime() {
118         return knownRuntime;
119     }
120 
121     public String runtimeBuilderClassName() {
122         return runtimeBuilderClassName;
123     }
124 
125     public static class Builder {
126         private @Nullable Context context;
127         private @Nullable List<String> warnings;
128 
129         public Builder withContext(Context context) {
130             this.context = context;
131             return this;
132         }
133 
134         public Builder withWarnings(List<String> warnings) {
135             this.warnings = warnings;
136             return this;
137         }
138 
139         public CalculatedContextValues build() {
140             return new CalculatedContextValues(this);
141         }
142     }
143 }
