16  package org.mybatis.generator.codegen.mybatis3;
17  
18  import java.util.List;
19  
20  import org.mybatis.generator.api.ProgressCallback;
21  import org.mybatis.generator.codegen.AbstractJavaClientGenerator;
22  import org.mybatis.generator.codegen.AbstractJavaGenerator;
23  import org.mybatis.generator.codegen.mybatis3.javamapper.SimpleAnnotatedClientGenerator;
24  import org.mybatis.generator.codegen.mybatis3.javamapper.SimpleJavaClientGenerator;
25  import org.mybatis.generator.codegen.mybatis3.model.SimpleModelGenerator;
26  import org.mybatis.generator.codegen.mybatis3.xmlmapper.SimpleXMLMapperGenerator;
27  import org.mybatis.generator.internal.ObjectFactory;
28  
29  /**
30   * Introspected table implementation for generating simple MyBatis3 artifacts (no "by example" methods, flat model,
31   * etc.)
32   *
33   * @author Jeff Butler
34   */
35  public class IntrospectedTableMyBatis3SimpleImpl extends IntrospectedTableMyBatis3Impl {
36      public IntrospectedTableMyBatis3SimpleImpl() {
37          super();
38      }
39  
40      @Override
41      protected void calculateXmlMapperGenerator(AbstractJavaClientGenerator javaClientGenerator, List<String> warnings,
42              ProgressCallback progressCallback) {
43          if (javaClientGenerator == null) {
44              if (context.getSqlMapGeneratorConfiguration() != null) {
45                  xmlMapperGenerator = new SimpleXMLMapperGenerator();
46              }
47          } else {
48              xmlMapperGenerator = javaClientGenerator.getMatchedXMLGenerator();
49          }
50  
51          initializeAbstractGenerator(xmlMapperGenerator, warnings, progressCallback);
52      }
53  
54      @Override
55      protected AbstractJavaClientGenerator createJavaClientGenerator() {
56          if (context.getJavaClientGeneratorConfiguration() == null) {
57              return null;
58          }
59  
60          String type = context.getJavaClientGeneratorConfiguration().getConfigurationType();
61  
62          AbstractJavaClientGenerator javaGenerator;
63          if ("XMLMAPPER".equalsIgnoreCase(type)) { //$NON-NLS-1$
64              javaGenerator = new SimpleJavaClientGenerator(getClientProject());
65          } else if ("ANNOTATEDMAPPER".equalsIgnoreCase(type)) { //$NON-NLS-1$
66              javaGenerator = new SimpleAnnotatedClientGenerator(getClientProject());
67          } else if ("MAPPER".equalsIgnoreCase(type)) { //$NON-NLS-1$
68              javaGenerator = new SimpleJavaClientGenerator(getClientProject());
69          } else {
70              javaGenerator = (AbstractJavaClientGenerator) ObjectFactory.createInternalObject(type);
71          }
72  
73          return javaGenerator;
74      }
75  
76      @Override
77      protected void calculateJavaModelGenerators(List<String> warnings, ProgressCallback progressCallback) {
78  
79          AbstractJavaGenerator javaGenerator = new SimpleModelGenerator(getModelProject());
80          initializeAbstractGenerator(javaGenerator, warnings, progressCallback);
81          javaGenerators.add(javaGenerator);
82      }
83  }
