51      @Override
52      public List<CompilationUnit> getCompilationUnits() {
53          FullyQualifiedTable table = introspectedTable.getFullyQualifiedTable();
54          progressCallback.startTask(getString(
55                  "Progress.8", table.toString())); //$NON-NLS-1$
56          Plugin plugins = context.getPlugins();
57          CommentGenerator commentGenerator = context.getCommentGenerator();
58  
59          FullyQualifiedJavaType type = new FullyQualifiedJavaType(
60                  introspectedTable.getBaseRecordType());
61          TopLevelClass topLevelClass = new TopLevelClass(type);
62          topLevelClass.setVisibility(JavaVisibility.PUBLIC);
63          commentGenerator.addJavaFileComment(topLevelClass);
64  
65          FullyQualifiedJavaType superClass = getSuperClass();
66          if (superClass != null) {
67              topLevelClass.setSuperClass(superClass);
68              topLevelClass.addImportedType(superClass);
69          }
70  
71          List<IntrospectedColumn> introspectedColumns = getColumnsInThisClass();
72  
73          if (introspectedTable.isConstructorBased()) {
74              addParameterizedConstructor(topLevelClass);
75              
76              if (!introspectedTable.isImmutable()) {
77                  addDefaultConstructor(topLevelClass);
78              }
79          }
80          
81          String rootClass = getRootClass();
82          for (IntrospectedColumn introspectedColumn : introspectedColumns) {
83              if (RootClassInfo.getInstance(rootClass, warnings)
84                      .containsProperty(introspectedColumn)) {
85                  continue;
86              }
87  
88              Field field = getJavaBeansField(introspectedColumn, context, introspectedTable);
89              if (plugins.modelFieldGenerated(field, topLevelClass,
90                      introspectedColumn, introspectedTable,
91                      Plugin.ModelClassType.BASE_RECORD)) {
92                  topLevelClass.addField(field);
93                  topLevelClass.addImportedType(field.getType());
94              }
95  
96              Method method = getJavaBeansGetter(introspectedColumn, context, introspectedTable);
97              if (plugins.modelGetterMethodGenerated(method, topLevelClass,
98                      introspectedColumn, introspectedTable,
99                      Plugin.ModelClassType.BASE_RECORD)) {
100                 topLevelClass.addMethod(method);
101             }
102 
103             if (!introspectedTable.isImmutable()) {
104                 method = getJavaBeansSetter(introspectedColumn, context, introspectedTable);
105                 if (plugins.modelSetterMethodGenerated(method, topLevelClass,
106                         introspectedColumn, introspectedTable,
107                         Plugin.ModelClassType.BASE_RECORD)) {
108                     topLevelClass.addMethod(method);
109                 }
110             }
111         }
112 
113         List<CompilationUnit> answer = new ArrayList<CompilationUnit>();
114         if (context.getPlugins().modelBaseRecordClassGenerated(
115                 topLevelClass, introspectedTable)) {
116             answer.add(topLevelClass);
117         }
118         return answer;
119     }

121     private FullyQualifiedJavaType getSuperClass() {
122         FullyQualifiedJavaType superClass;
123         if (introspectedTable.getRules().generatePrimaryKeyClass()) {
124             superClass = new FullyQualifiedJavaType(introspectedTable
125                     .getPrimaryKeyType());
126         } else {
127             String rootClass = getRootClass();
128             if (rootClass != null) {
129                 superClass = new FullyQualifiedJavaType(rootClass);
130             } else {
131                 superClass = null;
132             }
133         }
134 
135         return superClass;
136     }

138     private boolean includePrimaryKeyColumns() {
139         return !introspectedTable.getRules().generatePrimaryKeyClass()
140                 && introspectedTable.hasPrimaryKeyColumns();
141     }
142 
143     private boolean includeBLOBColumns() {
144         return !introspectedTable.getRules().generateRecordWithBLOBsClass()
145                 && introspectedTable.hasBLOBColumns();
146     }

148     private void addParameterizedConstructor(TopLevelClass topLevelClass) {
149         Method method = new Method();
150         method.setVisibility(JavaVisibility.PUBLIC);
151         method.setConstructor(true);
152         method.setName(topLevelClass.getType().getShortName());
153         context.getCommentGenerator().addGeneralMethodComment(method, introspectedTable);
154 
155         List<IntrospectedColumn> constructorColumns =
156             includeBLOBColumns() ? introspectedTable.getAllColumns() :
157                 introspectedTable.getNonBLOBColumns();
158             
159         for (IntrospectedColumn introspectedColumn : constructorColumns) {
160             method.addParameter(new Parameter(introspectedColumn.getFullyQualifiedJavaType(),
161                     introspectedColumn.getJavaProperty()));
162             topLevelClass.addImportedType(introspectedColumn.getFullyQualifiedJavaType());
163         }
164         
165         StringBuilder sb = new StringBuilder();
166         if (introspectedTable.getRules().generatePrimaryKeyClass()) {
167             boolean comma = false;
168             sb.append("super("); //$NON-NLS-1$
169             for (IntrospectedColumn introspectedColumn : introspectedTable
170                     .getPrimaryKeyColumns()) {
171                 if (comma) {
172                     sb.append(", "); //$NON-NLS-1$
173                 } else {
174                     comma = true;
175                 }
176                 sb.append(introspectedColumn.getJavaProperty());
177             }
178             sb.append(");"); //$NON-NLS-1$
179             method.addBodyLine(sb.toString());
180         }
181 
182         List<IntrospectedColumn> introspectedColumns = getColumnsInThisClass();
183         
184         for (IntrospectedColumn introspectedColumn : introspectedColumns) {
185             sb.setLength(0);
186             sb.append("this."); //$NON-NLS-1$
187             sb.append(introspectedColumn.getJavaProperty());
188             sb.append(" = "); //$NON-NLS-1$
189             sb.append(introspectedColumn.getJavaProperty());
190             sb.append(';');
191             method.addBodyLine(sb.toString());
192         }
193 
194         topLevelClass.addMethod(method);
195     }
196     
197     private List<IntrospectedColumn> getColumnsInThisClass() {
198         List<IntrospectedColumn> introspectedColumns;
199         if (includePrimaryKeyColumns()) {
200             if (includeBLOBColumns()) {
201                 introspectedColumns = introspectedTable.getAllColumns();
202             } else {
203                 introspectedColumns = introspectedTable.getNonBLOBColumns();
204             }
205         } else {
206             if (includeBLOBColumns()) {
207                 introspectedColumns = introspectedTable
208                         .getNonPrimaryKeyColumns();
209             } else {
210                 introspectedColumns = introspectedTable.getBaseColumns();
211             }
212         }
213         
214         return introspectedColumns;
215     }
