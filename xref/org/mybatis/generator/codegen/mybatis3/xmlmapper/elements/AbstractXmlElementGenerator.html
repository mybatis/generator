42      /**
43       * This method should return an XmlElement for the select key used to
44       * automatically generate keys.
45       *
46       * @param introspectedColumn
47       *            the column related to the select key statement
48       * @param generatedKey
49       *            the generated key for the current table
50       * @return the selectKey element
51       */
52      protected XmlElement getSelectKey(IntrospectedColumn introspectedColumn,
53              GeneratedKey generatedKey) {
54          String identityColumnType = introspectedColumn.getFullyQualifiedJavaType().getFullyQualifiedName();
55  
56          XmlElement answer = new XmlElement("selectKey"); //$NON-NLS-1$
57          answer.addAttribute(new Attribute("resultType", identityColumnType)); //$NON-NLS-1$
58          answer.addAttribute(new Attribute(
59                  "keyProperty", introspectedColumn.getJavaProperty())); //$NON-NLS-1$
60          answer.addAttribute(new Attribute("order", generatedKey.getMyBatis3Order())); //$NON-NLS-1$
61  
62          answer.addElement(new TextElement(generatedKey.getRuntimeSqlStatement()));
63  
64          return answer;
65      }

80          XmlElement ifElement = new XmlElement("if"); //$NON-NLS-1$
81          ifElement.addAttribute(new Attribute("test", "_parameter != null")); //$NON-NLS-1$ //$NON-NLS-2$

102     protected List<TextElement> buildSelectList(List<IntrospectedColumn> columns) {
103         return buildSelectList("", columns); //$NON-NLS-1$
104     }

106     protected List<TextElement> buildSelectList(String initial, List<IntrospectedColumn> columns) {
107         List<TextElement> answer = new ArrayList<>();
108         StringBuilder sb = new StringBuilder(initial);
109         Iterator<IntrospectedColumn> iter = columns.iterator();
110         while (iter.hasNext()) {
111             sb.append(MyBatis3FormattingUtilities.getSelectListPhrase(iter.next()));
112 
113             if (iter.hasNext()) {
114                 sb.append(", "); //$NON-NLS-1$
115             }
116 
117             if (sb.length() > 80) {
118                 answer.add(new TextElement(sb.toString()));
119                 sb.setLength(0);
120             }
121         }
122 
123         if (sb.length() > 0) {
124             answer.add(new TextElement(sb.toString()));
125         }
126 
127         return answer;
128     }

130     protected List<TextElement> buildPrimaryKeyWhereClause() {
131         List<TextElement> answer = new ArrayList<>();
132         boolean first = true;
133         for (IntrospectedColumn introspectedColumn : introspectedTable.getPrimaryKeyColumns()) {
134             String line;
135             if (first) {
136                 line = "where "; //$NON-NLS-1$
137                 first = false;
138             } else {
139                 line = "  and "; //$NON-NLS-1$
140             }
141 
142             line += MyBatis3FormattingUtilities.getEscapedColumnName(introspectedColumn);
143             line += " = "; //$NON-NLS-1$
144             line += MyBatis3FormattingUtilities.getParameterClause(introspectedColumn);
145             answer.add(new TextElement(line));
146         }
147 
148         return answer;
149     }

151     protected XmlElement buildInitialInsert(String statementId, FullyQualifiedJavaType parameterType) {
152         XmlElement answer = new XmlElement("insert"); //$NON-NLS-1$
153 
154         answer.addAttribute(new Attribute("id", statementId)); //$NON-NLS-1$
155 
156         answer.addAttribute(new Attribute("parameterType", parameterType.getFullyQualifiedName())); //$NON-NLS-1$
157 
158         context.getCommentGenerator().addComment(answer);
159 
160         introspectedTable.getGeneratedKey().ifPresent(gk ->
161                 introspectedTable.getColumn(gk.getColumn()).ifPresent(introspectedColumn -> {
162                     // if the column is null, then it's a configuration error. The
163                     // warning has already been reported
164                     if (gk.isJdbcStandard()) {
165                         answer.addAttribute(new Attribute("useGeneratedKeys", "true")); //$NON-NLS-1$ //$NON-NLS-2$
166                         answer.addAttribute(
167                                 new Attribute("keyProperty", introspectedColumn.getJavaProperty())); //$NON-NLS-1$
168                         answer.addAttribute(
169                                 new Attribute("keyColumn", introspectedColumn.getActualColumnName())); //$NON-NLS-1$
170                     } else {
171                         answer.addElement(getSelectKey(introspectedColumn, gk));
172                     }
173                 })
174         );
175 
176         return answer;
177     }

190     protected List<XmlElement> buildResultMapItems(ResultElementType elementType, List<IntrospectedColumn> columns) {
191         List<XmlElement> answer = new ArrayList<>();
192         for (IntrospectedColumn introspectedColumn : columns) {
193             XmlElement resultElement = new XmlElement(elementType.value);
194 
195             resultElement.addAttribute(buildColumnAttribute(introspectedColumn));
196             resultElement.addAttribute(new Attribute("property", introspectedColumn.getJavaProperty())); //$NON-NLS-1$
197             resultElement.addAttribute(new Attribute("jdbcType", introspectedColumn.getJdbcTypeName())); //$NON-NLS-1$
198 
199             if (stringHasValue(introspectedColumn.getTypeHandler())) {
200                 resultElement.addAttribute(
201                         new Attribute("typeHandler", introspectedColumn.getTypeHandler())); //$NON-NLS-1$
202             }
203 
204             answer.add(resultElement);
205         }
206 
207         return answer;
208     }

210     protected XmlElement buildConstructorElement(boolean includeBlobColumns) {
211         XmlElement constructor = new XmlElement("constructor"); //$NON-NLS-1$
212 
213         for (IntrospectedColumn introspectedColumn : introspectedTable.getPrimaryKeyColumns()) {
214             XmlElement resultElement = new XmlElement("idArg"); //$NON-NLS-1$
215 
216             resultElement.addAttribute(buildColumnAttribute(introspectedColumn));
217             resultElement.addAttribute(new Attribute("jdbcType", //$NON-NLS-1$
218                     introspectedColumn.getJdbcTypeName()));
219             resultElement.addAttribute(new Attribute("javaType", //$NON-NLS-1$
220                     introspectedColumn.getFullyQualifiedJavaType().getFullyQualifiedName()));
221 
222             if (stringHasValue(introspectedColumn.getTypeHandler())) {
223                 resultElement.addAttribute(
224                         new Attribute("typeHandler", introspectedColumn.getTypeHandler())); //$NON-NLS-1$
225             }
226 
227             constructor.addElement(resultElement);
228         }
229 
230         List<IntrospectedColumn> columns;
231         if (includeBlobColumns) {
232             columns = introspectedTable.getNonPrimaryKeyColumns();
233         } else {
234             columns = introspectedTable.getBaseColumns();
235         }
236         for (IntrospectedColumn introspectedColumn : columns) {
237             XmlElement resultElement = new XmlElement("arg"); //$NON-NLS-1$
238 
239             resultElement.addAttribute(buildColumnAttribute(introspectedColumn));
240             resultElement.addAttribute(new Attribute("jdbcType", //$NON-NLS-1$
241                     introspectedColumn.getJdbcTypeName()));
242 
243             if (introspectedColumn.getFullyQualifiedJavaType().isPrimitive()) {
244                 // need to use the MyBatis type alias for a primitive byte
245                 String s = '_'
246                         + introspectedColumn.getFullyQualifiedJavaType().getShortName();
247                 resultElement.addAttribute(new Attribute("javaType", s)); //$NON-NLS-1$
248             } else if ("byte[]".equals(introspectedColumn.getFullyQualifiedJavaType() //$NON-NLS-1$
249                     .getFullyQualifiedName())) {
250                 // need to use the MyBatis type alias for a primitive byte arry
251                 resultElement.addAttribute(new Attribute("javaType", //$NON-NLS-1$
252                         "_byte[]")); //$NON-NLS-1$
253             } else {
254                 resultElement.addAttribute(new Attribute("javaType", //$NON-NLS-1$
255                         introspectedColumn.getFullyQualifiedJavaType().getFullyQualifiedName()));
256             }
257 
258             if (stringHasValue(introspectedColumn.getTypeHandler())) {
259                 resultElement.addAttribute(new Attribute(
260                         "typeHandler", introspectedColumn.getTypeHandler())); //$NON-NLS-1$
261             }
262 
263             constructor.addElement(resultElement);
264         }
265 
266         return constructor;
267     }

274     protected XmlElement buildUpdateByExampleElement(String statementId, List<IntrospectedColumn> columns) {
275         XmlElement answer = new XmlElement("update"); //$NON-NLS-1$
276 
277         answer.addAttribute(new Attribute("id", statementId)); //$NON-NLS-1$
278 
279         answer.addAttribute(new Attribute("parameterType", "map")); //$NON-NLS-1$ //$NON-NLS-2$
280         context.getCommentGenerator().addComment(answer);
281 
282         StringBuilder sb = new StringBuilder();
283         sb.append("update "); //$NON-NLS-1$
284         sb.append(introspectedTable.getAliasedFullyQualifiedTableNameAtRuntime());
285         answer.addElement(new TextElement(sb.toString()));
286 
287         // set up for first column
288         sb.setLength(0);
289         sb.append("set "); //$NON-NLS-1$
290 
291         Iterator<IntrospectedColumn> iter = ListUtilities.removeGeneratedAlwaysColumns(columns).iterator();
292         while (iter.hasNext()) {
293             IntrospectedColumn introspectedColumn = iter.next();
294 
295             sb.append(MyBatis3FormattingUtilities.getAliasedEscapedColumnName(introspectedColumn));
296             sb.append(" = "); //$NON-NLS-1$
297             sb.append(MyBatis3FormattingUtilities.getParameterClause(introspectedColumn, "row.")); //$NON-NLS-1$
298 
299             if (iter.hasNext()) {
300                 sb.append(',');
301             }
302 
303             answer.addElement(new TextElement(sb.toString()));
304 
305             // set up for the next column
306             if (iter.hasNext()) {
307                 sb.setLength(0);
308                 OutputUtilities.xmlIndent(sb, 1);
309             }
310         }
311 
312         answer.addElement(getUpdateByExampleIncludeElement());
313         return answer;
314     }

316     protected XmlElement buildUpdateByPrimaryKeyElement(String statementId, String parameterType,
317                                                         List<IntrospectedColumn> columns) {
318         XmlElement answer = new XmlElement("update"); //$NON-NLS-1$
319 
320         answer.addAttribute(new Attribute("id", statementId)); //$NON-NLS-1$
321         answer.addAttribute(new Attribute("parameterType", parameterType)); //$NON-NLS-1$
322 
323         context.getCommentGenerator().addComment(answer);
324 
325         StringBuilder sb = new StringBuilder();
326         sb.append("update "); //$NON-NLS-1$
327         sb.append(introspectedTable.getFullyQualifiedTableNameAtRuntime());
328         answer.addElement(new TextElement(sb.toString()));
329 
330         // set up for first column
331         sb.setLength(0);
332         sb.append("set "); //$NON-NLS-1$
333 
334         Iterator<IntrospectedColumn> iter = ListUtilities.removeGeneratedAlwaysColumns(columns).iterator();
335         while (iter.hasNext()) {
336             IntrospectedColumn introspectedColumn = iter.next();
337 
338             sb.append(MyBatis3FormattingUtilities.getEscapedColumnName(introspectedColumn));
339             sb.append(" = "); //$NON-NLS-1$
340             sb.append(MyBatis3FormattingUtilities.getParameterClause(introspectedColumn));
341 
342             if (iter.hasNext()) {
343                 sb.append(',');
344             }
345 
346             answer.addElement(new TextElement(sb.toString()));
347 
348             // set up for the next column
349             if (iter.hasNext()) {
350                 sb.setLength(0);
351                 OutputUtilities.xmlIndent(sb, 1);
352             }
353         }
354 
355         buildPrimaryKeyWhereClause().forEach(answer::addElement);
356 
357         return answer;
358     }
