153             errors.add(getString("ValidationError.8", id)); //$NON-NLS-1$
154         } else {

156         }
157

159             javaClientGeneratorConfiguration.validate(errors, id);
160         }
161 
162         IntrospectedTable it = null;
163         try {
164             it = ObjectFactory.createIntrospectedTableForValidation(this);
165         } catch (Exception e) {
166             errors.add(getString("ValidationError.25", id)); //$NON-NLS-1$
167         }
168         
169         if (it != null && it.requiresXMLGenerator()) {
170             if (sqlMapGeneratorConfiguration == null) {
171                 errors.add(getString("ValidationError.9", id)); //$NON-NLS-1$
172             } else {
173                 sqlMapGeneratorConfiguration.validate(errors, id);
174             }
175         }
176 
177         if (tableConfigurations.size() == 0) {
178             errors.add(getString("ValidationError.3", id)); //$NON-NLS-1$
179         } else {
180             for (int i = 0; i < tableConfigurations.size(); i++) {
181                 TableConfiguration tc = tableConfigurations.get(i);
182 
183                 tc.validate(errors, i);
184             }
185         }
186 
187         for (PluginConfiguration pluginConfiguration : pluginConfigurations) {
188             pluginConfiguration.validate(errors, id);
189         }

386     public int getIntrospectionSteps() {
387         int steps = 0;
388 
389         steps++; // connect to database

393         // 1. Create introspected table implementation

395         steps += tableConfigurations.size() * 1;

397         return steps;

400     /**
401      * Introspect tables based on the configuration specified in the
402      * constructor. This method is long running.
403      * 
404      * @param callback
405      *            a progress callback if progress information is desired, or
406      *            <code>null</code>
407      * @param warnings
408      *            any warning generated from this method will be added to the
409      *            List. Warnings are always Strings.
410      * @param fullyQualifiedTableNames
411      *            a set of table names to generate. The elements of the set must
412      *            be Strings that exactly match what's specified in the
413      *            configuration. For example, if table name = "foo" and schema =
414      *            "bar", then the fully qualified table name is "foo.bar". If
415      *            the Set is null or empty, then all tables in the configuration
416      *            will be used for code generation.
417      * 
418      * @throws SQLException
419      *             if some error arises while introspecting the specified
420      *             database tables.
421      * @throws InterruptedException
422      *             if the progress callback reports a cancel
423      */
424     public void introspectTables(ProgressCallback callback,
425             List<String> warnings, Set<String> fullyQualifiedTableNames)
426             throws SQLException, InterruptedException {
427 
428         introspectedTables = new ArrayList<IntrospectedTable>();
429         JavaTypeResolver javaTypeResolver = ObjectFactory
430                 .createJavaTypeResolver(this, warnings);
431 
432         Connection connection = null;
433 
434         try {
435             callback.startTask(getString("Progress.0")); //$NON-NLS-1$
436             connection = getConnection();
437 
438             DatabaseIntrospector databaseIntrospector = new DatabaseIntrospector(
439                     this, connection.getMetaData(), javaTypeResolver, warnings);
440 
441             for (TableConfiguration tc : tableConfigurations) {
442                 String tableName = composeFullyQualifiedTableName(tc.getCatalog(), tc
443                                 .getSchema(), tc.getTableName(), '.');
444 
445                 if (fullyQualifiedTableNames != null
446                         && fullyQualifiedTableNames.size() > 0) {
447                     if (!fullyQualifiedTableNames.contains(tableName)) {
448                         continue;
449                     }
450                 }
451 
452                 if (!tc.areAnyStatementsEnabled()) {
453                     warnings.add(getString("Warning.0", tableName)); //$NON-NLS-1$
454                     continue;
455                 }
456 
457                 callback.startTask(getString("Progress.1", tableName)); //$NON-NLS-1$
458                 List<IntrospectedTable> tables = databaseIntrospector
459                         .introspectTables(tc);
460 
461                 if (tables != null) {
462                     introspectedTables.addAll(tables);
463                 }
464 
465                 callback.checkCancel();
466             }
467         } finally {
468             closeConnection(connection);
469         }
470     }

472     public int getGenerationSteps() {
473         int steps = 0;

484     public void generateFiles(ProgressCallback callback,
485             List<GeneratedJavaFile> generatedJavaFiles,
486             List<GeneratedXmlFile> generatedXmlFiles, List<String> warnings)
487             throws InterruptedException {
488 
489         pluginAggregator = new PluginAggregator();
490         for (PluginConfiguration pluginConfiguration : pluginConfigurations) {
491             Plugin plugin = ObjectFactory.createPlugin(this,
492                     pluginConfiguration);
493             if (plugin.validate(warnings)) {
494                 pluginAggregator.addPlugin(plugin);
495             } else {
496                 warnings.add(getString("Warning.24", //$NON-NLS-1$
497                         pluginConfiguration.getConfigurationType(), id));
498             }
499         }
500 
501         if (introspectedTables != null) {
502             for (IntrospectedTable introspectedTable : introspectedTables) {
503                 callback.checkCancel();
504 
505                 introspectedTable.initialize();
506                 introspectedTable.calculateGenerators(warnings, callback);
507                 generatedJavaFiles.addAll(introspectedTable
508                         .getGeneratedJavaFiles());
509                 generatedXmlFiles.addAll(introspectedTable
510                         .getGeneratedXmlFiles());
511 
512                 generatedJavaFiles.addAll(pluginAggregator
513                         .contextGenerateAdditionalJavaFiles(introspectedTable));
514                 generatedXmlFiles.addAll(pluginAggregator
515                         .contextGenerateAdditionalXmlFiles(introspectedTable));
516             }
517         }
518 
519         generatedJavaFiles.addAll(pluginAggregator
520                 .contextGenerateAdditionalJavaFiles());
521         generatedXmlFiles.addAll(pluginAggregator
522                 .contextGenerateAdditionalXmlFiles());
523     }
