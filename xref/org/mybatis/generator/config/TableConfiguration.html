131     public boolean isColumnIgnored(String columnName) {
132         for (Map.Entry<IgnoredColumn, Boolean> entry : ignoredColumns
133                 .entrySet()) {
134             IgnoredColumn ic = entry.getKey();
135             if (ic.isColumnNameDelimited()) {
136                 if (columnName.equals(ic.getColumnName())) {
137                     entry.setValue(Boolean.TRUE);
138                     return true;
139                 }
140             } else {
141                 if (columnName.equalsIgnoreCase(ic.getColumnName())) {
142                     entry.setValue(Boolean.TRUE);
143                     return true;
144                 }
145             }
146         }
147 
148         return false;
149     }

487     public void validate(List<String> errors, int listPosition) {
488         if (!stringHasValue(tableName)) {
489             errors.add(getString(
490                     "ValidationError.6", Integer.toString(listPosition))); //$NON-NLS-1$
491         }
492 
493         String fqTableName = composeFullyQualifiedTableName(
494                 catalog, schema, tableName, '.');
495 
496         if (generatedKey != null) {
497             generatedKey.validate(errors, fqTableName);
498         }
499 
500         if (isTrue(getProperty(PropertyRegistry.TABLE_USE_COLUMN_INDEXES))) {
501             // when using column indexes, either both or neither query ids
502             // should be set
503             if (selectByExampleStatementEnabled
504                     && selectByPrimaryKeyStatementEnabled) {
505                 boolean queryId1Set = stringHasValue(selectByExampleQueryId);
506                 boolean queryId2Set = stringHasValue(selectByPrimaryKeyQueryId);
507 
508                 if (queryId1Set != queryId2Set) {
509                     errors.add(getString("ValidationError.13", //$NON-NLS-1$
510                             fqTableName));
511                 }
512             }
513         }
514 
515         if (columnRenamingRule != null) {
516             columnRenamingRule.validate(errors, fqTableName);
517         }
518 
519         for (ColumnOverride columnOverride : columnOverrides) {
520             columnOverride.validate(errors, fqTableName);
521         }
522 
523         for (IgnoredColumn ignoredColumn : ignoredColumns.keySet()) {
524             ignoredColumn.validate(errors, fqTableName);
525         }
526     }
