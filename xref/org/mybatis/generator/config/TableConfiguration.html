221     /**
222      * Checks if is column ignored.
223      *
224      * @param columnName
225      *            the column name
226      * @return true, if is column ignored
227      */
228     public boolean isColumnIgnored(String columnName) {
229         for (Map.Entry<IgnoredColumn, Boolean> entry : ignoredColumns
230                 .entrySet()) {
231             if (entry.getKey().matches(columnName)) {
232                 entry.setValue(Boolean.TRUE);
233                 return true;
234             }
235         }
236         
237         for (IgnoredColumnPattern ignoredColumnPattern : ignoredColumnPatterns) {
238             if (ignoredColumnPattern.matches(columnName)) {
239                 return true;
240             }
241         }
242 
243         return false;
244     }

648             xmlElement.addAttribute(new Attribute(

726             for (ColumnOverride columnOverride : columnOverrides) {
727                 xmlElement.addElement(columnOverride.toXmlElement());
728             }
729         }
730 
731         return xmlElement;

802     /**
803      * Validate.
804      *
805      * @param errors
806      *            the errors
807      * @param listPosition
808      *            the list position
809      */
810     public void validate(List<String> errors, int listPosition) {
811         if (!stringHasValue(tableName)) {
812             errors.add(getString(
813                     "ValidationError.6", Integer.toString(listPosition))); //$NON-NLS-1$
814         }
815 
816         String fqTableName = composeFullyQualifiedTableName(
817                 catalog, schema, tableName, '.');
818 
819         if (generatedKey != null) {
820             generatedKey.validate(errors, fqTableName);
821         }
822 
823         // when using column indexes, either both or neither query ids
824         // should be set
825         if (isTrue(getProperty(PropertyRegistry.TABLE_USE_COLUMN_INDEXES))
826                 && selectByExampleStatementEnabled
827                 && selectByPrimaryKeyStatementEnabled) {
828             boolean queryId1Set = stringHasValue(selectByExampleQueryId);
829             boolean queryId2Set = stringHasValue(selectByPrimaryKeyQueryId);
830 
831             if (queryId1Set != queryId2Set) {
832                 errors.add(getString("ValidationError.13", //$NON-NLS-1$
833                         fqTableName));
834             }
835         }
836 
837         if (columnRenamingRule != null) {
838             columnRenamingRule.validate(errors, fqTableName);
839         }
840 
841         for (ColumnOverride columnOverride : columnOverrides) {
842             columnOverride.validate(errors, fqTableName);
843         }
844 
845         for (IgnoredColumn ignoredColumn : ignoredColumns.keySet()) {
846             ignoredColumn.validate(errors, fqTableName);
847         }
848 
849         for (IgnoredColumnPattern ignoredColumnPattern : ignoredColumnPatterns) {
850             ignoredColumnPattern.validate(errors, fqTableName);
851         }
852     }
