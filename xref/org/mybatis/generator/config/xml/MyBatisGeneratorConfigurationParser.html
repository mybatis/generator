56      public MyBatisGeneratorConfigurationParser(Properties properties) {
57          super();
58          if (properties == null) {
59              this.properties = System.getProperties();
60          } else {
61              this.properties = properties;
62          }

65      public Configuration parseConfiguration(Element rootNode)
66              throws XMLParserException {
67  
68          Configuration configuration = new Configuration();
69  
70          NodeList nodeList = rootNode.getChildNodes();
71          for (int i = 0; i < nodeList.getLength(); i++) {
72              Node childNode = nodeList.item(i);
73  
74              if (childNode.getNodeType() != Node.ELEMENT_NODE) {
75                  continue;
76              }
77  
78              if ("properties".equals(childNode.getNodeName())) { //$NON-NLS-1$
79                  parseProperties(configuration, childNode);
80              } else if ("classPathEntry".equals(childNode.getNodeName())) { //$NON-NLS-1$
81                  parseClassPathEntry(configuration, childNode);
82              } else if ("context".equals(childNode.getNodeName())) { //$NON-NLS-1$
83                  parseContext(configuration, childNode);
84              }
85          }
86  
87          return configuration;
88      }

90      private void parseProperties(Configuration configuration, Node node)
91              throws XMLParserException {
92          Properties attributes = parseAttributes(node);
93          String resource = attributes.getProperty("resource"); //$NON-NLS-1$
94          String url = attributes.getProperty("url"); //$NON-NLS-1$
95  
96          if (!stringHasValue(resource)
97                  && !stringHasValue(url)) {
98              throw new XMLParserException(getString("RuntimeError.14")); //$NON-NLS-1$
99          }
100 
101         if (stringHasValue(resource)
102                 && stringHasValue(url)) {
103             throw new XMLParserException(getString("RuntimeError.14")); //$NON-NLS-1$
104         }
105 
106         URL resourceUrl;
107 
108         try {
109             if (stringHasValue(resource)) {
110                 resourceUrl = ObjectFactory.getResource(resource);
111                 if (resourceUrl == null) {
112                     throw new XMLParserException(getString(
113                             "RuntimeError.15", resource)); //$NON-NLS-1$
114                 }
115             } else {
116                 resourceUrl = new URL(url);
117             }
118 
119             InputStream inputStream = resourceUrl.openConnection()
120                     .getInputStream();
121 
122             properties.load(inputStream);
123             inputStream.close();
124         } catch (IOException e) {
125             if (stringHasValue(resource)) {
126                 throw new XMLParserException(getString(
127                         "RuntimeError.16", resource)); //$NON-NLS-1$
128             } else {
129                 throw new XMLParserException(getString(
130                         "RuntimeError.17", url)); //$NON-NLS-1$
131             }
132         }
133     }

135     private void parseContext(Configuration configuration, Node node) {
136 
137         Properties attributes = parseAttributes(node);
138         String defaultModelType = attributes.getProperty("defaultModelType"); //$NON-NLS-1$
139         String targetRuntime = attributes.getProperty("targetRuntime"); //$NON-NLS-1$
140         String introspectedColumnImpl = attributes
141                 .getProperty("introspectedColumnImpl"); //$NON-NLS-1$
142         String id = attributes.getProperty("id"); //$NON-NLS-1$
143 
144         ModelType mt = defaultModelType == null ? null : ModelType
145                 .getModelType(defaultModelType);
146 
147         Context context = new Context(mt);
148         context.setId(id);
149         if (stringHasValue(introspectedColumnImpl)) {
150             context.setIntrospectedColumnImpl(introspectedColumnImpl);
151         }
152         if (stringHasValue(targetRuntime)) {
153             context.setTargetRuntime(targetRuntime);
154         }
155 
156         configuration.addContext(context);
157 
158         NodeList nodeList = node.getChildNodes();
159         for (int i = 0; i < nodeList.getLength(); i++) {
160             Node childNode = nodeList.item(i);
161 
162             if (childNode.getNodeType() != Node.ELEMENT_NODE) {
163                 continue;
164             }
165 
166             if ("property".equals(childNode.getNodeName())) { //$NON-NLS-1$
167                 parseProperty(context, childNode);
168             } else if ("plugin".equals(childNode.getNodeName())) { //$NON-NLS-1$
169                 parsePlugin(context, childNode);
170             } else if ("commentGenerator".equals(childNode.getNodeName())) { //$NON-NLS-1$
171                 parseCommentGenerator(context, childNode);
172             } else if ("jdbcConnection".equals(childNode.getNodeName())) { //$NON-NLS-1$
173                 parseJdbcConnection(context, childNode);
174             } else if ("javaModelGenerator".equals(childNode.getNodeName())) { //$NON-NLS-1$
175                 parseJavaModelGenerator(context, childNode);
176             } else if ("javaTypeResolver".equals(childNode.getNodeName())) { //$NON-NLS-1$
177                 parseJavaTypeResolver(context, childNode);
178             } else if ("sqlMapGenerator".equals(childNode.getNodeName())) { //$NON-NLS-1$
179                 parseSqlMapGenerator(context, childNode);
180             } else if ("javaClientGenerator".equals(childNode.getNodeName())) { //$NON-NLS-1$
181                 parseJavaClientGenerator(context, childNode);
182             } else if ("table".equals(childNode.getNodeName())) { //$NON-NLS-1$
183                 parseTable(context, childNode);
184             }
185         }
186     }

272

274             tc.setSelectByPrimaryKeyStatementEnabled(
275                     isTrue(enableSelectByPrimaryKey));

277 
278         if (stringHasValue(enableSelectByExample)) {
279             tc.setSelectByExampleStatementEnabled(

282 
283         if (stringHasValue(enableUpdateByPrimaryKey)) {

318         }

321             tc.setWildcardEscapingEnabled(isTrue(escapeWildcards));
322         }

324         if (stringHasValue(delimitIdentifiers)) {
325             tc.setDelimitIdentifiers(isTrue(delimitIdentifiers));
326         }
327 
328         if (stringHasValue(delimitAllColumns)) {
329             tc.setAllColumnDelimitingEnabled(isTrue(delimitAllColumns));

333         for (int i = 0; i < nodeList.getLength(); i++) {
334             Node childNode = nodeList.item(i);
335 
336             if (childNode.getNodeType() != Node.ELEMENT_NODE) {
337                 continue;

348             } else if ("columnRenamingRule".equals(childNode.getNodeName())) { //$NON-NLS-1$
349                 parseColumnRenamingRule(tc, childNode);
350             }
351         }

354     private void parseColumnOverride(TableConfiguration tc, Node node) {
355         Properties attributes = parseAttributes(node);
356         String column = attributes.getProperty("column"); //$NON-NLS-1$
357         String property = attributes.getProperty("property"); //$NON-NLS-1$
358         String javaType = attributes.getProperty("javaType"); //$NON-NLS-1$
359         String jdbcType = attributes.getProperty("jdbcType"); //$NON-NLS-1$
360         String typeHandler = attributes.getProperty("typeHandler"); //$NON-NLS-1$
361         String delimitedColumnName = attributes
362                 .getProperty("delimitedColumnName"); //$NON-NLS-1$
363 
364         ColumnOverride co = new ColumnOverride(column);
365 
366         if (stringHasValue(property)) {
367             co.setJavaProperty(property);
368         }
369 
370         if (stringHasValue(javaType)) {
371             co.setJavaType(javaType);
372         }
373 
374         if (stringHasValue(jdbcType)) {
375             co.setJdbcType(jdbcType);
376         }
377 
378         if (stringHasValue(typeHandler)) {
379             co.setTypeHandler(typeHandler);
380         }
381 
382         if (stringHasValue(delimitedColumnName)) {
383             co.setColumnNameDelimited(isTrue(delimitedColumnName));
384         }
385 
386         NodeList nodeList = node.getChildNodes();
387         for (int i = 0; i < nodeList.getLength(); i++) {
388             Node childNode = nodeList.item(i);
389 
390             if (childNode.getNodeType() != Node.ELEMENT_NODE) {
391                 continue;
392             }
393 
394             if ("property".equals(childNode.getNodeName())) { //$NON-NLS-1$
395                 parseProperty(co, childNode);
396             }
397         }
398 
399         tc.addColumnOverride(co);
400     }

607     private Properties parseAttributes(Node node) {
608         Properties attributes = new Properties();
609         NamedNodeMap nnm = node.getAttributes();
610         for (int i = 0; i < nnm.getLength(); i++) {
611             Node attribute = nnm.item(i);
612             String value = parsePropertyTokens(attribute.getNodeValue());
613             attributes.put(attribute.getNodeName(), value);
614         }
615 
616         return attributes;
617     }

619     private String parsePropertyTokens(String string) {
620         final String OPEN = "${"; //$NON-NLS-1$
621         final String CLOSE = "}"; //$NON-NLS-1$
622 
623         String newString = string;
624         if (newString != null) {
625             int start = newString.indexOf(OPEN);
626             int end = newString.indexOf(CLOSE);
627 
628             while (start > -1 && end > start) {
629                 String prepend = newString.substring(0, start);
630                 String append = newString.substring(end + CLOSE.length());
631                 String propName = newString.substring(start + OPEN.length(),
632                         end);
633                 String propValue = properties.getProperty(propName);
634                 if (propValue != null) {
635                     newString = prepend + propValue + append;
636                 }
637 
638                 start = newString.indexOf(OPEN, end);
639                 end = newString.indexOf(CLOSE, end);
640             }
641         }
642 
643         return newString;
644     }
