66      /**
67       * Adds a suitable comment to warn users that the element was generated, and
68       * when it was generated.
69       *
70       * @param xmlElement the xml element
71       */
72      @Override
73      public void addComment(XmlElement xmlElement) {
74          if (suppressAllComments) {
75              return;
76          }
77  
78          xmlElement.addElement(new TextElement("<!--")); //$NON-NLS-1$
79  
80          StringBuilder sb = new StringBuilder();
81          sb.append("  WARNING - "); //$NON-NLS-1$
82          sb.append(MergeConstants.NEW_ELEMENT_TAG);
83          xmlElement.addElement(new TextElement(sb.toString()));
84          xmlElement.addElement(
85                  new TextElement("  This element is automatically generated by MyBatis Generator," //$NON-NLS-1$
86                          + " do not modify.")); //$NON-NLS-1$
87  
88          String s = getDateString();
89          if (s != null) {
90              sb.setLength(0);
91              sb.append("  This element was generated on "); //$NON-NLS-1$
92              sb.append(s);
93              sb.append('.');
94              xmlElement.addElement(new TextElement(sb.toString()));
95          }
96  
97          xmlElement.addElement(new TextElement("-->")); //$NON-NLS-1$
98      }

100     @Override
101     public void addConfigurationProperties(Properties props) {
102         this.properties.putAll(props);
103 
104         suppressDate = isTrue(properties.getProperty(PropertyRegistry.COMMENT_GENERATOR_SUPPRESS_DATE));
105 
106         suppressAllComments = isTrue(properties.getProperty(PropertyRegistry.COMMENT_GENERATOR_SUPPRESS_ALL_COMMENTS));
107 
108         addRemarkComments = isTrue(properties.getProperty(PropertyRegistry.COMMENT_GENERATOR_ADD_REMARK_COMMENTS));
109 
110         String dateFormatString = properties.getProperty(PropertyRegistry.COMMENT_GENERATOR_DATE_FORMAT);
111         if (StringUtility.stringHasValue(dateFormatString)) {
112             dateFormat = new SimpleDateFormat(dateFormatString);
113         }
114     }

116     /**
117      * This method adds the custom javadoc tag for. You may do nothing if you do not
118      * wish to include the Javadoc tag - however, if you do not include the Javadoc
119      * tag then the Java merge capability of the eclipse plugin will break.
120      *
121      * @param javaElement       the java element
122      * @param markAsDoNotDelete the mark as do not delete
123      */
124     protected void addJavadocTag(JavaElement javaElement, boolean markAsDoNotDelete) {
125         javaElement.addJavaDocLine(" *"); //$NON-NLS-1$
126         StringBuilder sb = new StringBuilder();
127         sb.append(" * "); //$NON-NLS-1$
128         sb.append(MergeConstants.NEW_ELEMENT_TAG);
129         if (markAsDoNotDelete) {
130             sb.append(" do_not_delete_during_merge"); //$NON-NLS-1$
131         }
132         String s = getDateString();
133         if (s != null) {
134             sb.append(' ');
135             sb.append(s);
136         }
137         javaElement.addJavaDocLine(sb.toString());
138     }
139 
140     /**
141      * Returns a formated date string to include in the Javadoc tag and XML
142      * comments. You may return null if you do not want the date in these
143      * documentation elements.
144      *
145      * @return a string representing the current timestamp, or null
146      */
147     protected String getDateString() {
148         if (suppressDate) {
149             return null;
150         } else if (dateFormat != null) {
151             return dateFormat.format(new Date());
152         } else {
153             return new Date().toString();
154         }
155     }

197     @Override
198     public void addModelClassComment(TopLevelClass topLevelClass, IntrospectedTable introspectedTable) {
199         if (suppressAllComments || !addRemarkComments) {
200             return;
201         }
202 
203         topLevelClass.addJavaDocLine("/**"); //$NON-NLS-1$
204 
205         String remarks = introspectedTable.getRemarks();
206         if (addRemarkComments && StringUtility.stringHasValue(remarks)) {
207             topLevelClass.addJavaDocLine(" * Database Table Remarks:"); //$NON-NLS-1$
208             String[] remarkLines = remarks.split(System.getProperty("line.separator")); //$NON-NLS-1$
209             for (String remarkLine : remarkLines) {
210                 topLevelClass.addJavaDocLine(" *   " + remarkLine); //$NON-NLS-1$
211             }
212         }
213         topLevelClass.addJavaDocLine(" *"); //$NON-NLS-1$
214 
215         topLevelClass.addJavaDocLine(" * This class was generated by MyBatis Generator."); //$NON-NLS-1$
216 
217         String s = " * This class corresponds to the database table " //$NON-NLS-1$
218                 + introspectedTable.getFullyQualifiedTable();
219         topLevelClass.addJavaDocLine(s);
220 
221         topLevelClass.addJavaDocLine(" */"); //$NON-NLS-1$
222     }

440     private String getGeneratedAnnotation(String comment) {
441         StringBuilder buffer = new StringBuilder();
442         buffer.append("@Generated("); //$NON-NLS-1$
443         if (suppressAllComments) {
444             buffer.append('\"');
445         } else {
446             buffer.append("value=\""); //$NON-NLS-1$
447         }
448 
449         buffer.append(MyBatisGenerator.class.getName());
450         buffer.append('\"');
451 
452         if (!suppressDate && !suppressAllComments) {
453             buffer.append(", date=\""); //$NON-NLS-1$
454             buffer.append(DateTimeFormatter.ISO_OFFSET_DATE_TIME.format(ZonedDateTime.now()));
455             buffer.append('\"');
456         }
457 
458         if (!suppressAllComments) {
459             buffer.append(", comments=\""); //$NON-NLS-1$
460             buffer.append(comment);
461             buffer.append('\"');
462         }
463 
464         buffer.append(')');
465         return buffer.toString();
466     }
