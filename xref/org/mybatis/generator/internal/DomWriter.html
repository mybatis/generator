38  /**
39   * This class is used to generate a String representation of an XML document. It
40   * is very much based on the class dom.Writer from the Apache Xerces examples,
41   * but I've simplified and updated it.
42   * 
43   * @author Andy Clark, IBM (Original work)
44   * @author Jeff Butler (derivation)
45   */
46  public class DomWriter {

55      public synchronized String toString(Document document)
56              throws ShellException {
57          StringWriter sw = new StringWriter();
58          printWriter = new PrintWriter(sw);
59          write(document);
60          String s = sw.toString();
61          return s;
62      }

64      /** Returns a sorted list of attributes. */
65      protected Attr[] sortAttributes(NamedNodeMap attrs) {
66  
67          int len = (attrs != null) ? attrs.getLength() : 0;
68          Attr array[] = new Attr[len];
69          for (int i = 0; i < len; i++) {
70              array[i] = (Attr) attrs.item(i);
71          }
72          for (int i = 0; i < len - 1; i++) {
73              String name = array[i].getNodeName();
74              int index = i;
75              for (int j = i + 1; j < len; j++) {
76                  String curName = array[j].getNodeName();
77                  if (curName.compareTo(name) < 0) {
78                      name = curName;
79                      index = j;
80                  }
81              }
82              if (index != i) {
83                  Attr temp = array[i];
84                  array[i] = array[index];
85                  array[index] = temp;
86              }
87          }
88  
89          return array;
90  
91      }

93      /** Normalizes and prints the given string. */
94      protected void normalizeAndPrint(String s, boolean isAttValue) {
95  
96          int len = (s != null) ? s.length() : 0;
97          for (int i = 0; i < len; i++) {
98              char c = s.charAt(i);
99              normalizeAndPrint(c, isAttValue);
100         }
101 
102     }

104     /** Normalizes and print the given character. */
105     protected void normalizeAndPrint(char c, boolean isAttValue) {
106 
107         switch (c) {
108         case '<': {
109             printWriter.print("&lt;"); //$NON-NLS-1$
110             break;
111         }
112         case '>': {
113             printWriter.print("&gt;"); //$NON-NLS-1$
114             break;
115         }
116         case '&': {
117             printWriter.print("&amp;"); //$NON-NLS-1$
118             break;
119         }
120         case '"': {
121             // A '"' that appears in character data
122             // does not need to be escaped.
123             if (isAttValue) {
124                 printWriter.print("&quot;"); //$NON-NLS-1$
125             } else {
126                 printWriter.print('"');
127             }
128             break;
129         }
130         case '\r': {
131             // If CR is part of the document's content, it
132             // must not be printed as a literal otherwise
133             // it would be normalized to LF when the document
134             // is reparsed.
135             printWriter.print("&#xD;"); //$NON-NLS-1$
136             break;
137         }
138         default: {
139             // In XML 1.1, control chars in the ranges [#x1-#x1F, #x7F-#x9F]
140             // must be escaped.
141             //
142             // Escape space characters that would be normalized to #x20 in
143             // attribute values
144             // when the document is reparsed.
145             //
146             // Escape NEL (0x85) and LSEP (0x2028) that appear in content
147             // if the document is XML 1.1, since they would be normalized to LF
148             // when the document is reparsed.
149             if (isXML11
150                     && ((c >= 0x01 && c <= 0x1F && c != 0x09 && c != 0x0A)
151                             || (c >= 0x7F && c <= 0x9F) || c == 0x2028)
152                     || isAttValue && (c == 0x09 || c == 0x0A)) {
153                 printWriter.print("&#x"); //$NON-NLS-1$
154                 printWriter.print(Integer.toHexString(c).toUpperCase());
155                 printWriter.print(';');
156             } else {
157                 printWriter.print(c);
158             }
159         }
160         }
161     }

163     /** Extracts the XML version from the Document. */
164     protected String getVersion(Document document) {
165         if (document == null) {
166             return null;
167         }
168         String version = null;
169         Method getXMLVersion = null;
170         try {
171             getXMLVersion = document.getClass().getMethod("getXmlVersion", //$NON-NLS-1$
172                     new Class[] {});
173             // If Document class implements DOM L3, this method will exist.
174             if (getXMLVersion != null) {
175                 version = (String) getXMLVersion.invoke(document,
176                         (Object[]) null);
177             }
178         } catch (Exception e) {
179             // Either this locator object doesn't have
180             // this method, or we're on an old JDK.
181         }
182         return version;
183     }

185     protected void writeAnyNode(Node node) throws ShellException {
186         // is there anything to do?
187         if (node == null) {
188             return;
189         }
190 
191         short type = node.getNodeType();
192         switch (type) {
193         case Node.DOCUMENT_NODE:
194             write((Document) node);
195             break;
196 
197         case Node.DOCUMENT_TYPE_NODE:
198             write((DocumentType) node);
199             break;
200 
201         case Node.ELEMENT_NODE:
202             write((Element) node);
203             break;
204 
205         case Node.ENTITY_REFERENCE_NODE:
206             write((EntityReference) node);
207             break;
208 
209         case Node.CDATA_SECTION_NODE:
210             write((CDATASection) node);
211             break;
212 
213         case Node.TEXT_NODE:
214             write((Text) node);
215             break;
216 
217         case Node.PROCESSING_INSTRUCTION_NODE:
218             write((ProcessingInstruction) node);
219             break;
220 
221         case Node.COMMENT_NODE:
222             write((Comment) node);
223             break;
224 
225         default:
226             throw new ShellException(getString(
227                     "RuntimeError.18", Short.toString(type))); //$NON-NLS-1$
228         }
229     }

231     protected void write(Document node) throws ShellException {
232         isXML11 = "1.1".equals(getVersion(node)); //$NON-NLS-1$
233         if (isXML11) {
234             printWriter.println("<?xml version=\"1.1\" encoding=\"UTF-8\"?>"); //$NON-NLS-1$
235         } else {
236             printWriter.println("<?xml version=\"1.0\" encoding=\"UTF-8\"?>"); //$NON-NLS-1$
237         }
238         printWriter.flush();
239         write(node.getDoctype());
240         write(node.getDocumentElement());
241     }
242 
243     protected void write(DocumentType node) throws ShellException {
244         printWriter.print("<!DOCTYPE "); //$NON-NLS-1$
245         printWriter.print(node.getName());
246         String publicId = node.getPublicId();
247         String systemId = node.getSystemId();
248         if (publicId != null) {
249             printWriter.print(" PUBLIC \""); //$NON-NLS-1$
250             printWriter.print(publicId);
251             printWriter.print("\" \""); //$NON-NLS-1$
252             printWriter.print(systemId);
253             printWriter.print('\"');
254         } else if (systemId != null) {
255             printWriter.print(" SYSTEM \""); //$NON-NLS-1$
256             printWriter.print(systemId);
257             printWriter.print('"');
258         }
259 
260         String internalSubset = node.getInternalSubset();
261         if (internalSubset != null) {
262             printWriter.println(" ["); //$NON-NLS-1$
263             printWriter.print(internalSubset);
264             printWriter.print(']');
265         }
266         printWriter.println('>');
267     }

269     protected void write(Element node) throws ShellException {
270         printWriter.print('<');
271         printWriter.print(node.getNodeName());
272         Attr attrs[] = sortAttributes(node.getAttributes());
273         for (Attr attr : attrs) {
274             printWriter.print(' ');
275             printWriter.print(attr.getNodeName());
276             printWriter.print("=\""); //$NON-NLS-1$
277             normalizeAndPrint(attr.getNodeValue(), true);
278             printWriter.print('"');
279         }
280 
281         if (node.getChildNodes().getLength() == 0) {
282             printWriter.print(" />"); //$NON-NLS-1$
283             printWriter.flush();
284         } else {
285             printWriter.print('>');
286             printWriter.flush();
287 
288             Node child = node.getFirstChild();
289             while (child != null) {
290                 writeAnyNode(child);
291                 child = child.getNextSibling();
292             }
293 
294             printWriter.print("</"); //$NON-NLS-1$
295             printWriter.print(node.getNodeName());
296             printWriter.print('>');
297             printWriter.flush();
298         }
299     }

301     protected void write(EntityReference node) {
302         printWriter.print('&');
303         printWriter.print(node.getNodeName());
304         printWriter.print(';');
305         printWriter.flush();
306     }

308     protected void write(CDATASection node) {
309         printWriter.print("<![CDATA["); //$NON-NLS-1$
310         printWriter.print(node.getNodeValue());
311         printWriter.print("]]>"); //$NON-NLS-1$
312         printWriter.flush();
313     }
314 
315     protected void write(Text node) {
316         normalizeAndPrint(node.getNodeValue(), false);
317         printWriter.flush();
318     }
319 
320     protected void write(ProcessingInstruction node) {
321         printWriter.print("<?"); //$NON-NLS-1$
322         printWriter.print(node.getNodeName());
323         String data = node.getNodeValue();
324         if (data != null && data.length() > 0) {
325             printWriter.print(' ');
326             printWriter.print(data);
327         }
328         printWriter.print("?>"); //$NON-NLS-1$
329         printWriter.flush();
330     }
331 
332     protected void write(Comment node) {
333         printWriter.print("<!--"); //$NON-NLS-1$
334         String comment = node.getNodeValue();
335         if (comment != null && comment.length() > 0) {
336             printWriter.print(comment);
337         }
338         printWriter.print("-->"); //$NON-NLS-1$
339         printWriter.flush();
340     }
