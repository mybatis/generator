16  package org.mybatis.generator.logging;
17  
18  import static org.mybatis.generator.internal.util.messages.Messages.getString;
19  
20  import org.mybatis.generator.logging.commons.JakartaCommonsLoggingLogFactory;
21  import org.mybatis.generator.logging.jdk14.Jdk14LoggingLogFactory;
22  import org.mybatis.generator.logging.log4j.Log4jLoggingLogFactory;
23  import org.mybatis.generator.logging.log4j2.Log4j2LoggingLogFactory;
24  import org.mybatis.generator.logging.nologging.NoLoggingLogFactory;
25  import org.mybatis.generator.logging.slf4j.Slf4jLoggingLogFactory;
26  
27  /**
28   * Factory for creating loggers
29   * 
30   * @author Jeff Butler
31   * 
32   */
33  public class LogFactory {
34      private static AbstractLogFactory logFactory;
35      public static String MARKER = "MYBATIS-GENERATOR"; //$NON-NLS-1$
36  
37      static {
38          tryImplementation(new Slf4jLoggingLogFactory());
39          tryImplementation(new JakartaCommonsLoggingLogFactory());
40          tryImplementation(new Log4j2LoggingLogFactory());
41          tryImplementation(new Log4jLoggingLogFactory());
42          tryImplementation(new Jdk14LoggingLogFactory());
43          tryImplementation(new NoLoggingLogFactory());
44      }
45  
46      public static Log getLog(Class<?> clazz) {
47          try {
48              return logFactory.getLog(clazz);
49          } catch (Throwable t) {
50              throw new RuntimeException(getString("RuntimeError.21", //$NON-NLS-1$
51                      clazz.getName(), t.getMessage()), t);
52          }
53      }
54  
55      /**
56       * This method will switch the logging implementation to Java native
57       * logging. This is useful in situations where you want to use Java native
58       * logging to log activity but Log4J is on the classpath. Note that this
59       * method is only effective for log classes obtained after calling this
60       * method. If you intend to use this method you should call it before
61       * calling any other method.
62       */
63      public static synchronized void forceJavaLogging() {
64          setImplementation(new Jdk14LoggingLogFactory());
65      }
66  
67      public static synchronized void forceSlf4jLogging() {
68          setImplementation(new Slf4jLoggingLogFactory());
69      }
70  
71      public static synchronized void forceCommonsLogging() {
72          setImplementation(new JakartaCommonsLoggingLogFactory());
73      }
74  
75      public static synchronized void forceLog4jLogging() {
76          setImplementation(new Log4jLoggingLogFactory());
77      }
78  
79      public static synchronized void forceLog4j2Logging() {
80          setImplementation(new Log4j2LoggingLogFactory());
81      }
82  
83      public static synchronized void forceNoLogging() {
84          setImplementation(new NoLoggingLogFactory());
85      }
86  
87      public static void setLogFactory(AbstractLogFactory logFactory) {
88          setImplementation(logFactory);
89      }
90  
91      private static void tryImplementation(AbstractLogFactory factory) {
92          if (logFactory == null) {
93              try {
94                  setImplementation(factory);
95              } catch (LogException e) {
96                  // ignore
97              }
98          }
99      }
100 
101     private static void setImplementation(AbstractLogFactory factory) {
102         try {
103             Log log = factory.getLog(LogFactory.class);
104             if (log.isDebugEnabled()) {
105                 log.debug("Logging initialized using '" + factory + "' adapter."); //$NON-NLS-1$ //$NON-NLS-2$
106             }
107             logFactory = factory;
108         } catch (Throwable t) {
109             throw new LogException("Error setting Log implementation.  Cause: " + t.getMessage(), t); //$NON-NLS-1$
110         }
111     }
112 }
