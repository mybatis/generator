16  package org.mybatis.generator.merge.java;
17  
18  import com.github.javaparser.printer.configuration.DefaultConfigurationOption;
19  import com.github.javaparser.printer.configuration.DefaultPrinterConfiguration;
20  import com.github.javaparser.printer.configuration.Indentation;
21  import com.github.javaparser.printer.configuration.imports.EclipseImportOrderingStrategy;
22  
23  /**
24   * Printer configuration that orders imports in "Eclipse" order (statics first, etc.).
25   */
26  public class EclipseOrderedPrinterConfiguration extends DefaultPrinterConfiguration {
27      public EclipseOrderedPrinterConfiguration() {
28          addOption(new DefaultConfigurationOption(DefaultPrinterConfiguration.ConfigOption.ORDER_IMPORTS, true));
29          addOption(new DefaultConfigurationOption(DefaultPrinterConfiguration.ConfigOption.SORT_IMPORTS_STRATEGY,
30                          new EclipseImportOrderingStrategy()));
31          addOption(new DefaultConfigurationOption(ConfigOption.PRINT_COMMENTS, Boolean.TRUE));
32          addOption(new DefaultConfigurationOption(ConfigOption.PRINT_JAVADOC, Boolean.TRUE));
33          addOption(new DefaultConfigurationOption(ConfigOption.SPACE_AROUND_OPERATORS, Boolean.TRUE));
34  
35          addOption(new DefaultConfigurationOption(ConfigOption.INDENT_CASE_IN_SWITCH, false));
36          addOption(new DefaultConfigurationOption(ConfigOption.MAX_ENUM_CONSTANTS_TO_ALIGN_HORIZONTALLY, 999));
37          addOption(new DefaultConfigurationOption(ConfigOption.INDENTATION,
38                  new Indentation(Indentation.IndentType.SPACES, 4)));
39          addOption(new DefaultConfigurationOption(ConfigOption.INDENT_PRINT_ARRAYS_OF_ANNOTATIONS, true));
40      }
41  }
