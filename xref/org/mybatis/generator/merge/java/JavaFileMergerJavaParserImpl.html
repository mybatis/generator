35  /**
36   * This class handles the task of merging changes into an existing Java file using JavaParser.
37   * It supports merging by removing methods and fields that have specific Javadoc tags or annotations.
38   *
39   * <p>Given an existing source file and a newly generated file of the same name, the merger will:
40   * <ol>
41   *     <li>Parse the existing file looking for custom additions. A custom addition is defined in these ways:
42   *         <ul>
43   *             <li>A body element (field, method, nested class, etc.) not marked as generated - missing both a
44   *                 <code>@Generated</code> annotation and an older style custom Javadoc tag.</li>
45   *             <li>A body element (field, method, nested class, etc.) marked as generated by an older style custom
46   *                 Javadoc tag and also containing the phrase "do_not_delete_during_merge".</li>
47   *             <li>Any import in the existing file that is not present in the newly generated file</li>
48   *             <li>Any super interface in the existing file that is not present in the newly generated file</li>
49   *             <li>Any enum constant missing a "generated" marker in the existing file that is not present in the new
50   *                 file</li>
51   *         </ul>
52   *         It is important to know that the parser will only look for direct children of either the public type or the
53   *         first non-public type in a source file.
54   *     </li>
55   *     <li>If there are no custom additions, the newly generated file is returned unmodified.</li>
56   *     <li>If there are custom additions, then:
57   *         <ul>
58   *             <li>Add any imports present in the existing file but missing in the new file</li>
59   *             <li>Remove any members in the new file that match custom additions in the existing file</li>
60   *             <li>Add all custom additions from the existing file</li>
61   *             <li>The merged file is formatted and returned.</li>
62   *         </ul>
63   *     </li>
64   * </ol>
65   *
66   * <p>This implementation differs from the original Eclipse-based implementation in the following ways:</p>
67   * <ol>
68   *     <li>This implementation supports merging enums and records</li>
69   *     <li>This implementation supports merging when the existing file is a class or interface, and the newly generated
70   *         file is a record (the result will be a record).
71   *     </li>
72   *     <li>This implementation does not support merging the super class from an existing file to the newly generated
73   *         file.</li>
74   *     <li>This implementation does not attempt to preserve custom annotations added to generated elements. With the
75   *         generator now generating code with many annotations, it is challenging to distinguish between annotations
76   *         created by MBG, and custom annotations added after code generation by a user. If you need to add
77   *         annotations to generated elements, consider implementing a plugin that will create the annotations whenever
78   *         the generator runs.
79   *     </li>
80   * </ol>
81   *
82   * @author Freeman (original)
83   * @author Jeff Butler (refactoring and enhancements)
84   */
85  public class JavaFileMergerJavaParserImpl implements JavaFileMerger {
86      private final Printer printer;
87  
88      public JavaFileMergerJavaParserImpl(JavaMergerFactory.PrinterConfiguration printerConfiguration) {
89          printer = switch (printerConfiguration) {
90          case ECLIPSE -> new DefaultPrettyPrinter(new EclipseOrderedPrinterConfiguration());
91          case LEXICAL_PRESERVING -> new DefaultLexicalPreservingPrinter();
92          };
93      }
94  
95      /**
96       * Merge a newly generated Java file with existing Java file content.
97       *
98       * @param newFileContent the content of the newly generated Java file
99       * @param existingFileContent the content of the existing Java file
100      * @return the merged source, properly formatted
101      * @throws MergeException if the file cannot be merged for some reason
102      */
103     @Override
104     public String getMergedSource(String newFileContent, String existingFileContent) throws MergeException {
105         ParserConfiguration parserConfiguration = new ParserConfiguration();
106         parserConfiguration.setLanguageLevel(ParserConfiguration.LanguageLevel.JAVA_25);
107         parserConfiguration.setLexicalPreservationEnabled(true);
108         JavaParser javaParser = new JavaParser(parserConfiguration);
109 
110         ParseResults existingFileParseResults = parseAndFindMainTypeDeclaration(javaParser, existingFileContent,
111                 FileType.EXISTING_FILE);
112 
113         // Gather custom members from the existing file. If none, just return the new file as is
114         CustomMemberGatherer customMemberGatherer = new CustomMemberGatherer(existingFileParseResults.typeDeclaration);
115         if (!customMemberGatherer.hasAnyMembersToMerge()) {
116             return newFileContent;
117         }
118 
119         // Custom members exist, need to merge...
120         ParseResults newFileParseResults = parseAndFindMainTypeDeclaration(javaParser, newFileContent,
121                 FileType.NEW_FILE);
122 
123         // Delete elements in the new file that match doNotDeleteMembers from the existing file
124         customMemberGatherer.doNotDeleteBodyMembers()
125                 .forEach(m -> deleteDuplicateMemberIfExists(newFileParseResults.typeDeclaration, m));
126 
127         // Look for custom imports in the existing file and merge into the new file
128         JavaMergeUtilities
129                 .findCustomImports(existingFileParseResults.compilationUnit, newFileParseResults.compilationUnit)
130                 .forEach(newFileParseResults.compilationUnit::addImport);
131 
132         // Add custom body members from the existing file to the new file
133         customMemberGatherer.allCustomBodyMembers().forEach(newFileParseResults.typeDeclaration::addMember);
134 
135         // Add custom enum constants from the existing file to the new file
136         if (newFileParseResults.typeDeclaration.isEnumDeclaration()) {
137             customMemberGatherer.customEnumConstants()
138                     .forEach(newFileParseResults.typeDeclaration.asEnumDeclaration()::addEntry);
139         }
140 
141         // Look for custom super interfaces in the existing file and merge into the new file
142         JavaMergeUtilities
143                 .findCustomSuperInterfaces(existingFileParseResults.typeDeclaration,
144                         newFileParseResults.typeDeclaration)
145                 .forEach(t -> JavaMergeUtilities.addSuperInterface(newFileParseResults.typeDeclaration, t));
146 
147         // Return the new (merged) file
148         return printer.print(newFileParseResults.compilationUnit);
149     }
150 
151     private ParseResults parseAndFindMainTypeDeclaration(JavaParser javaParser, String source, FileType fileType)
152             throws MergeException {
153         ParseResult<CompilationUnit> parseResult = javaParser.parse(source);
154 
155         // little hack to pull the result out of the lambda. This allows us to avoid "orElseThrow()" later on
156         @Nullable CompilationUnit[] compilationUnits = new CompilationUnit [1];
157         parseResult.ifSuccessful(cu -> compilationUnits[0] = cu);
158 
159         if (compilationUnits[0] == null) {
160             List<String> details = parseResult.getProblems().stream()
161                     .map(Problem::toString)
162                     .toList();
163             throw new MergeException(getString("RuntimeError.28", fileType.toString()), details); //$NON-NLS-1$
164         }
165 
166         return new ParseResults(compilationUnits[0], findMainTypeDeclaration(compilationUnits[0], fileType));
167     }
168 
169     private void deleteDuplicateMemberIfExists(TypeDeclaration<?> newTypeDeclaration,
170                                                       BodyDeclaration<?> member) {
171         newTypeDeclaration.getMembers().stream()
172                 .filter(td -> JavaMergeUtilities.membersMatch(td, member))
173                 .findFirst()
174                 .ifPresent(newTypeDeclaration::remove);
175     }
176 
177     private TypeDeclaration<?> findMainTypeDeclaration(CompilationUnit compilationUnit, FileType fileType)
178             throws MergeException {
179         // Return the first public type declaration, or the first type declaration if no public one exists
180         TypeDeclaration<?> firstType = null;
181         for (TypeDeclaration<?> typeDeclaration : compilationUnit.getTypes()) {
182             if (firstType == null) {
183                 firstType = typeDeclaration;
184             }
185             if (typeDeclaration.isPublic()) {
186                 return typeDeclaration;
187             }
188         }
189         if (firstType == null) {
190             throw new MergeException(getString("RuntimeError.29", fileType.toString())); //$NON-NLS-1$
191         }
192         return firstType;
193     }
194 
195     private record ParseResults(CompilationUnit compilationUnit, TypeDeclaration<?> typeDeclaration) {}
196 
197     private enum FileType {
198         NEW_FILE("new Java file"), //$NON-NLS-1$
199         EXISTING_FILE("existing Java file"); //$NON-NLS-1$
200 
201         private final String displayText;
202 
203         FileType(String displayText) {
204             this.displayText = displayText;
205         }
206 
207         @Override
208         public String toString() {
209             return displayText;
210         }
211     }
