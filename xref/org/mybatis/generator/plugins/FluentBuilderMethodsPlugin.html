16  package org.mybatis.generator.plugins;
17  
18  import java.util.List;
19  
20  import org.mybatis.generator.api.IntrospectedColumn;
21  import org.mybatis.generator.api.IntrospectedTable;
22  import org.mybatis.generator.api.IntrospectedTable.TargetRuntime;
23  import org.mybatis.generator.api.PluginAdapter;
24  import org.mybatis.generator.api.dom.java.JavaVisibility;
25  import org.mybatis.generator.api.dom.java.Method;
26  import org.mybatis.generator.api.dom.java.TopLevelClass;
27  
28  /**
29   * This plugin adds fluent builder methods to the generated model classes.
30   *
31   * <p>Example:
32   *
33   * <p>Given the domain class <code>MyDomainClass</code> with setter-method <code>setValue(Object v)</code>
34   *
35   * <p>The plugin will create the additional Method <code>public MyDomainClass withValue(Object v)</code>
36   *
37   *
38   * @author Stefan Lack
39   */
40  public class FluentBuilderMethodsPlugin extends PluginAdapter {
41  
42      @Override
43      public boolean validate(List<String> warnings) {
44          return true;
45      }
46  
47      @Override
48      public boolean modelSetterMethodGenerated(Method method,
49              TopLevelClass topLevelClass, IntrospectedColumn introspectedColumn,
50              IntrospectedTable introspectedTable,
51              ModelClassType modelClassType) {
52  
53          Method fluentMethod = new Method("with" + method.getName().substring(3)); //$NON-NLS-1$
54          fluentMethod.setVisibility(JavaVisibility.PUBLIC);
55          fluentMethod.setReturnType(topLevelClass.getType());
56          fluentMethod.getParameters().addAll(method.getParameters());
57  
58          if (introspectedTable.getTargetRuntime() == TargetRuntime.MYBATIS3_DSQL) {
59              context.getCommentGenerator().addGeneralMethodAnnotation(fluentMethod,
60                      introspectedTable, topLevelClass.getImportedTypes());
61          } else {
62              context.getCommentGenerator().addGeneralMethodComment(fluentMethod,
63                      introspectedTable);
64          }
65  
66          String s = "this." //$NON-NLS-1$
67                  + method.getName()
68                  + '('
69                  + introspectedColumn.getJavaProperty()
70                  + ");"; //$NON-NLS-1$
71          fluentMethod.addBodyLine(s); //$NON-NLS-1$
72          fluentMethod.addBodyLine("return this;"); //$NON-NLS-1$
73  
74          topLevelClass.addMethod(fluentMethod);
75  
76          return super.modelSetterMethodGenerated(method, topLevelClass, introspectedColumn,
77                  introspectedTable, modelClassType);
78      }
79  }
