58      @Override
59      public List<CompilationUnit> getCompilationUnits() {
60          progressCallback.startTask(getString("Progress.17", //$NON-NLS-1$
61                  introspectedTable.getFullyQualifiedTable().toString()));
62          preCalculate();
63          
64          Interface interfaze = createBasicInterface();
65  
66          TopLevelClass supportClass = getSupportClass();
67          String staticImportString =
68                  supportClass.getType().getFullyQualifiedNameWithoutTypeParameters() + ".*"; //$NON-NLS-1$
69          interfaze.addStaticImport(staticImportString);
70  
71          addBasicCountMethod(interfaze);
72          addBasicDeleteMethod(interfaze);
73          addBasicInsertMethod(interfaze);
74          addBasicInsertMultipleMethod(interfaze);
75          if (introspectedTable.getGeneratedKey() != null) {
76              addBasicInsertMultipleHelperMethod(interfaze);
77          }
78          addBasicSelectOneMethod(interfaze);
79          addBasicSelectManyMethod(interfaze);
80          addBasicUpdateMethod(interfaze);
81          
82          addCountByExampleMethod(interfaze);
83          addDeleteByExampleMethod(interfaze);
84          addDeleteByPrimaryKeyMethod(interfaze);
85          addInsertOneMethod(interfaze);
86          addInsertMultipleMethod(interfaze);
87          addInsertSelectiveMethod(interfaze);
88          addSelectListField(interfaze);
89          addSelectByExampleMethod(interfaze);
90          addSelectDistinctByExampleMethod(interfaze);
91          addSelectByPrimaryKeyMethod(interfaze);
92          addUpdateByExampleMethod(interfaze);
93          addUpdateAllMethod(interfaze);
94          addUpdateSelectiveMethod(interfaze);
95          addUpdateByPrimaryKeyMethod(interfaze);
96          addUpdateByPrimaryKeySelectiveMethod(interfaze);
97          
98          List<CompilationUnit> answer = new ArrayList<>();
99          if (context.getPlugins().clientGenerated(interfaze, introspectedTable)) {
100             answer.add(interfaze);
101         }
102 
103         if (context.getPlugins().dynamicSqlSupportGenerated(supportClass, introspectedTable)) {
104             answer.add(supportClass);
105         }
106 
107         return answer;
108     }
