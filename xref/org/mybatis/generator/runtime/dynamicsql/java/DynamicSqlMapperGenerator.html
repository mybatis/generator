57  public class DynamicSqlMapperGenerator extends AbstractJavaGenerator {
58      // record type for insert, select, update
59      protected final FullyQualifiedJavaType recordType;
60      // id to use for the common result map
61      protected final String resultMapId;
62      // name of the field containing the table in the support class
63      protected final String tableFieldName;
64      protected final FragmentGenerator fragmentGenerator;
65      protected final boolean hasGeneratedKeys;
66  
67      public DynamicSqlMapperGenerator(Builder builder) {
68          super(builder);
69  
70          recordType = new FullyQualifiedJavaType(introspectedTable.getBaseRecordType());
71          resultMapId = recordType.getShortNameWithoutTypeArguments() + "Result"; //$NON-NLS-1$
72          boolean useSnakeCase = introspectedTable
73                  .findTableOrClientGeneratorPropertyAsBoolean(PropertyRegistry.ANY_USE_SNAKE_CASE_IDENTIFIERS);
74  
75          tableFieldName = calculateTableFieldName(useSnakeCase);
76          fragmentGenerator = new FragmentGenerator.Builder()
77                  .withIntrospectedTable(introspectedTable)
78                  .withResultMapId(resultMapId)
79                  .withTableFieldName(tableFieldName)
80                  .useSnakeCase(useSnakeCase)
81                  .withRecordType(recordType)
82                  .withCommentGenerator(commentGenerator)
83                  .build();
84  
85          hasGeneratedKeys = introspectedTable.getGeneratedKey().isPresent();
86      }
87  
88      private String calculateTableFieldName(boolean useSnakeCase) {
89          String name =
90                  JavaBeansUtil.getValidPropertyName(introspectedTable.getMyBatisDynamicSQLTableObjectName());
91          if (useSnakeCase) {
92              name = StringUtility.convertCamelCaseToSnakeCase(name);
93          }
94  
95          return name;
96      }

98      @Override
99      public List<CompilationUnit> getCompilationUnits() {
100         progressCallback.startTask(getString("Progress.17", //$NON-NLS-1$
101                 introspectedTable.getFullyQualifiedTable().toString()));
102 
103         Interface interfaze = createBasicInterface();
104 
105         TopLevelClass supportClass = getSupportClass();
106         String staticImportString =
107                 supportClass.getType().getFullyQualifiedNameWithoutTypeParameters() + ".*"; //$NON-NLS-1$
108         interfaze.addStaticImport(staticImportString);
109 
110         if (hasGeneratedKeys) {
111             addBasicInsertMethod(interfaze);
112             addBasicInsertMultipleMethod(interfaze);
113         }
114 
115         boolean reuseResultMap = addBasicSelectManyMethod(interfaze);
116         addBasicSelectOneMethod(interfaze, reuseResultMap);
117 
118         addGeneralCountMethod(interfaze);
119         addGeneralDeleteMethod(interfaze);
120         addDeleteByPrimaryKeyMethod(interfaze);
121         addInsertOneMethod(interfaze);
122         addInsertMultipleMethod(interfaze);
123         addInsertSelectiveMethod(interfaze);
124         addSelectListField(interfaze);
125         addGeneralSelectOneMethod(interfaze);
126         addGeneralSelectMethod(interfaze);
127         addSelectDistinctMethod(interfaze);
128         addSelectByPrimaryKeyMethod(interfaze);
129         addGeneralUpdateMethod(interfaze);
130         addUpdateAllMethod(interfaze);
131         addUpdateSelectiveMethod(interfaze);
132         addUpdateByPrimaryKeyMethod(interfaze);
133         addUpdateByPrimaryKeySelectiveMethod(interfaze);
134 
135         List<CompilationUnit> answer = new ArrayList<>();
136         if (pluginAggregator.clientGenerated(interfaze, introspectedTable)) {
137             answer.add(interfaze);
138         }
139 
140         if (pluginAggregator.dynamicSqlSupportGenerated(supportClass, introspectedTable)) {
141             answer.add(supportClass);
142         }
143 
144         return answer;
145     }

147     protected Interface createBasicInterface() {
148         FullyQualifiedJavaType type = new FullyQualifiedJavaType(introspectedTable.getMyBatis3JavaMapperType());
149         Interface interfaze = new Interface(type);
150         interfaze.setVisibility(JavaVisibility.PUBLIC);
151         commentGenerator.addJavaFileComment(interfaze);
152         interfaze.addImportedType(new FullyQualifiedJavaType("org.apache.ibatis.annotations.Mapper")); //$NON-NLS-1$
153         interfaze.addAnnotation("@Mapper"); //$NON-NLS-1$
154 
155         RootClassAndInterfaceUtility.addRootInterfaceIfNecessary(interfaze, introspectedTable);
156         return interfaze;
157     }
158 
159     protected TopLevelClass getSupportClass() {
160         return initializeSubBuilder(new DynamicSqlSupportClassGenerator.Builder())
161                 .build()
162                 .generate();
163     }

165     protected void addInsertOneMethod(Interface interfaze) {
166         var generated = initializeSubBuilder(new InsertMethodGenerator.Builder())
167                 .withTableFieldName(tableFieldName)
168                 .withRecordType(recordType)
169                 .withFragmentGenerator(fragmentGenerator)
170                 .build()
171                 .execute(interfaze);
172 
173         if (generated && !hasGeneratedKeys) {
174             // add common interface
175             addCommonInsertInterface(interfaze);
176         }
177     }
178 
179     protected void addCommonInsertInterface(Interface interfaze) {
180         FullyQualifiedJavaType superInterface = new FullyQualifiedJavaType(
181                 "org.mybatis.dynamic.sql.util.mybatis3.CommonInsertMapper<" //$NON-NLS-1$
182                         + recordType.getFullyQualifiedName() + ">"); //$NON-NLS-1$
183         interfaze.addSuperInterface(superInterface);
184         interfaze.addImportedTypes(superInterface.getImportList().stream()
185                 .map(FullyQualifiedJavaType::new)
186                 .collect(Collectors.toSet()));
187     }

240     protected void addSelectListField(Interface interfaze) {
241         var generator = initializeSubBuilder(new SelectListGenerator.Builder())
242                 .withFragmentGenerator(fragmentGenerator)
243                 .build();
244 
245         JavaFieldAndImports javaFieldAndImports = generator.generateFieldAndImports();
246 
247         if (generator.callPlugins(javaFieldAndImports.getField(), interfaze)) {
248             interfaze.addField(javaFieldAndImports.getField());
249             interfaze.addImportedTypes(javaFieldAndImports.getImports());
250         }
251     }
