59      public String calculateFieldName(String tableFieldName, IntrospectedColumn column) {
60          String fieldName = column.getJavaProperty();
61          if (useSnakeCase) {
62              fieldName = StringUtility.convertCamelCaseToSnakeCase(fieldName);
63          }
64  
65          if (fieldName.equals(tableFieldName)) {
66              // name collision, no shortcut generated
67              fieldName = tableFieldName + "." + fieldName; //$NON-NLS-1$
68          }
69          return fieldName;
70      }
71  
72      public String getSelectList() {
73          return introspectedTable.getAllColumns().stream()
74                  .map(c -> calculateFieldName(tableFieldName, c))
75                  .collect(Collectors.joining(", ")); //$NON-NLS-1$
76      }

78      public JavaMethodParts getPrimaryKeyWhereClauseAndParameters() {
79          JavaMethodParts.Builder builder = new JavaMethodParts.Builder();
80  
81          boolean first = true;
82          for (IntrospectedColumn column : introspectedTable.getPrimaryKeyColumns()) {
83              String parameterName = column.getJavaProperty();
84              if (!useSnakeCase) {
85                  parameterName += "_"; //$NON-NLS-1$
86              }
87  
88              String fieldName = calculateFieldName(tableFieldName, column);
89              builder.withImport(column.getFullyQualifiedJavaType());
90              builder.withParameter(new Parameter(column.getFullyQualifiedJavaType(), parameterName));
91              if (first) {
92                  builder.withBodyLine(javaIndent(1) + "c.where(" + fieldName //$NON-NLS-1$
93                          + ", isEqualTo(" + parameterName //$NON-NLS-1$
94                          + "))"); //$NON-NLS-1$
95                  first = false;
96              } else {
97                  builder.withBodyLine(javaIndent(1) + ".and(" + fieldName //$NON-NLS-1$
98                          + ", isEqualTo(" + parameterName //$NON-NLS-1$
99                          + "))"); //$NON-NLS-1$
100             }
101         }
102         builder.withBodyLine(");"); //$NON-NLS-1$
103 
104         return builder.build();
105     }

107     public List<String> getPrimaryKeyWhereClauseForUpdate(String prefix) {
108         List<String> lines = new ArrayList<>();
109 
110         boolean first = true;
111         for (IntrospectedColumn column : introspectedTable.getPrimaryKeyColumns()) {
112             String fieldName = calculateFieldName(tableFieldName, column);
113             String methodName = JavaBeansUtil.getCallingGetterMethodName(column);
114             if (first) {
115                 lines.add(prefix + ".where(" + fieldName //$NON-NLS-1$
116                         + ", isEqualTo(row::" + methodName //$NON-NLS-1$
117                         + "))"); //$NON-NLS-1$
118                 first = false;
119             } else {
120                 lines.add(prefix + ".and(" + fieldName //$NON-NLS-1$
121                         + ", isEqualTo(row::" + methodName //$NON-NLS-1$
122                         + "))"); //$NON-NLS-1$
123             }
124         }
125 
126         return lines;
127     }

129     public JavaMethodParts getAnnotatedConstructorArgs() {
130         JavaMethodParts.Builder builder = new JavaMethodParts.Builder();
131 
132         builder.withImport(new FullyQualifiedJavaType("org.apache.ibatis.type.JdbcType")); //$NON-NLS-1$
133         builder.withImport(new FullyQualifiedJavaType("org.apache.ibatis.annotations.Arg")); //$NON-NLS-1$
134         builder.withImport(new FullyQualifiedJavaType("org.apache.ibatis.annotations.Results")); //$NON-NLS-1$
135 
136         builder.withAnnotation("@Results(id=\"" + resultMapId + "\")"); //$NON-NLS-1$ //$NON-NLS-2$
137 
138         Set<FullyQualifiedJavaType> imports = new HashSet<>();
139         for (IntrospectedColumn introspectedColumn : introspectedTable.getPrimaryKeyColumns()) {
140             builder.withAnnotation(getArgAnnotation(imports, introspectedColumn, true));
141         }
142 
143         for (IntrospectedColumn introspectedColumn : introspectedTable.getNonPrimaryKeyColumns()) {
144             builder.withAnnotation(getArgAnnotation(imports, introspectedColumn, false));
145         }
146 
147         return builder.withImports(imports).build();
148     }

150     public JavaMethodParts getAnnotatedResults() {
151         JavaMethodParts.Builder builder = new JavaMethodParts.Builder();
152 
153         builder.withImport(new FullyQualifiedJavaType("org.apache.ibatis.type.JdbcType")); //$NON-NLS-1$
154         builder.withImport(new FullyQualifiedJavaType("org.apache.ibatis.annotations.Result")); //$NON-NLS-1$
155         builder.withImport(new FullyQualifiedJavaType("org.apache.ibatis.annotations.Results")); //$NON-NLS-1$
156 
157         builder.withAnnotation("@Results(id=\"" + resultMapId + "\", value = {"); //$NON-NLS-1$ //$NON-NLS-2$
158 
159         StringBuilder sb = new StringBuilder();
160 
161         Set<FullyQualifiedJavaType> imports = new HashSet<>();
162         Iterator<IntrospectedColumn> iterPk = introspectedTable.getPrimaryKeyColumns().iterator();
163         Iterator<IntrospectedColumn> iterNonPk = introspectedTable.getNonPrimaryKeyColumns().iterator();
164         while (iterPk.hasNext()) {
165             IntrospectedColumn introspectedColumn = iterPk.next();
166             sb.setLength(0);
167             javaIndent(sb, 1);
168             sb.append(getResultAnnotation(imports, introspectedColumn, true));
169 
170             if (iterPk.hasNext() || iterNonPk.hasNext()) {
171                 sb.append(',');
172             }
173 
174             builder.withAnnotation(sb.toString());
175         }
176 
177         while (iterNonPk.hasNext()) {
178             IntrospectedColumn introspectedColumn = iterNonPk.next();
179             sb.setLength(0);
180             javaIndent(sb, 1);
181             sb.append(getResultAnnotation(imports, introspectedColumn, false));
182 
183             if (iterNonPk.hasNext()) {
184                 sb.append(',');
185             }
186 
187             builder.withAnnotation(sb.toString());
188         }
189 
190         builder.withAnnotation("})") //$NON-NLS-1$
191                 .withImports(imports);
192 
193         return builder.build();
194     }

196     private String getArgAnnotation(Set<FullyQualifiedJavaType> imports, IntrospectedColumn introspectedColumn,
197                                     boolean idColumn) {
198         imports.add(introspectedColumn.getFullyQualifiedJavaType());
199 
200         return "@Arg(column=\"" //$NON-NLS-1$
201                 + introspectedColumn.getActualColumnName()
202                 + "\", javaType=" //$NON-NLS-1$
203                 + introspectedColumn.getFullyQualifiedJavaType().getShortName()
204                 + ".class" //$NON-NLS-1$
205                 + generateAdditionalItems(imports, introspectedColumn, idColumn)
206                 + ')';//$NON-NLS-1$
207     }
208 
209     private String getResultAnnotation(Set<FullyQualifiedJavaType> imports, IntrospectedColumn introspectedColumn,
210             boolean idColumn) {
211         return "@Result(column=\"" //$NON-NLS-1$
212                 + introspectedColumn.getActualColumnName()
213                 + "\", property=\"" //$NON-NLS-1$
214                 + introspectedColumn.getJavaProperty()
215                 + '\"'
216                 + generateAdditionalItems(imports, introspectedColumn, idColumn)
217                 + ')'; //$NON-NLS-1$
218     }
219 
220     private String generateAdditionalItems(Set<FullyQualifiedJavaType> imports, IntrospectedColumn introspectedColumn,
221                                            boolean idColumn) {
222         StringBuilder sb = new StringBuilder();
223 
224         introspectedColumn.getTypeHandler().map(FullyQualifiedJavaType::new).ifPresent(th -> {
225             imports.add(th);
226             sb.append(", typeHandler="); //$NON-NLS-1$
227             sb.append(th.getShortName());
228             sb.append(".class"); //$NON-NLS-1$
229         });
230 
231         sb.append(", jdbcType=JdbcType."); //$NON-NLS-1$
232         sb.append(introspectedColumn.getJdbcTypeName());
233         if (idColumn) {
234             sb.append(", id=true"); //$NON-NLS-1$
235         }
236 
237         return sb.toString();
238     }

253     private List<String> getSetLinesForUpdateStatement(List<IntrospectedColumn> columnList, String firstLinePrefix,
254                                                        String subsequentLinePrefix, boolean terminate,
255                                                        String fragment) {
256         List<String> lines = new ArrayList<>();
257         List<IntrospectedColumn> columns = ListUtilities.filterColumnsForUpdate(columnList);
258         Iterator<IntrospectedColumn> iter = columns.iterator();
259         boolean first = true;
260         while (iter.hasNext()) {
261             IntrospectedColumn column = iter.next();
262             String fieldName = calculateFieldName(tableFieldName, column);
263             String methodName = JavaBeansUtil.getCallingGetterMethodName(column);
264 
265             String start;
266             if (first) {
267                 start = firstLinePrefix;
268                 first = false;
269             } else {
270                 start = subsequentLinePrefix;
271             }
272 
273             String line = start
274                     + ".set(" //$NON-NLS-1$
275                     + fieldName
276                     + ")." //$NON-NLS-1$
277                     + fragment
278                     + "(row::" //$NON-NLS-1$
279                     + methodName
280                     + ")"; //$NON-NLS-1$
281 
282             if (terminate && !iter.hasNext()) {
283                 line += ";"; //$NON-NLS-1$
284             }
285 
286             lines.add(line);
287         }
288 
289         return lines;
290     }

292     public JavaMethodAndImports generateGeneralSelectMethod(boolean isDistinct) {
293         String methodName;
294         String utilMethodName;
295         if (isDistinct) {
296             methodName = "selectDistinct"; //$NON-NLS-1$
297             utilMethodName = "selectDistinct"; //$NON-NLS-1$
298         } else {
299             methodName = "select"; //$NON-NLS-1$
300             utilMethodName = "selectList"; //$NON-NLS-1$
301         }
302 
303         Set<FullyQualifiedJavaType> imports = new HashSet<>();
304 
305         FullyQualifiedJavaType parameterType = new FullyQualifiedJavaType(
306                 "org.mybatis.dynamic.sql.dsl.SelectDSLCompleter"); //$NON-NLS-1$
307 
308         imports.add(parameterType);
309         imports.add(new FullyQualifiedJavaType("org.mybatis.dynamic.sql.util.mybatis3.MyBatis3Utils")); //$NON-NLS-1$
310 
311         FullyQualifiedJavaType returnType = FullyQualifiedJavaType.getNewListInstance();
312         returnType.addTypeArgument(recordType);
313 
314         imports.add(returnType);
315 
316         Method method = new Method(methodName);
317         method.setDefault(true);
318         method.addParameter(new Parameter(parameterType, "completer")); //$NON-NLS-1$
319 
320         commentGenerator.addGeneralMethodAnnotation(method, introspectedTable, imports);
321 
322         method.setReturnType(returnType);
323         String line = String.format(
324                 "return MyBatis3Utils.%s(this::selectMany, selectList, %s, completer);", //$NON-NLS-1$
325                 utilMethodName, tableFieldName);
326         method.addBodyLine(line);
327 
328         return JavaMethodAndImports.withMethod(method)
329                 .withImports(imports)
330                 .build();
331     }
