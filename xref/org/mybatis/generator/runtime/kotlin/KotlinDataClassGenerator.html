16  package org.mybatis.generator.runtime.kotlin;
17  
18  import static org.mybatis.generator.internal.util.messages.Messages.getString;
19  
20  import java.util.Collections;
21  import java.util.List;
22  
23  import org.mybatis.generator.api.CommentGenerator;
24  import org.mybatis.generator.api.FullyQualifiedTable;
25  import org.mybatis.generator.api.IntrospectedColumn;
26  import org.mybatis.generator.api.dom.kotlin.FullyQualifiedKotlinType;
27  import org.mybatis.generator.api.dom.kotlin.JavaToKotlinTypeConverter;
28  import org.mybatis.generator.api.dom.kotlin.KotlinFile;
29  import org.mybatis.generator.api.dom.kotlin.KotlinModifier;
30  import org.mybatis.generator.api.dom.kotlin.KotlinProperty;
31  import org.mybatis.generator.api.dom.kotlin.KotlinType;
32  import org.mybatis.generator.codegen.AbstractKotlinGenerator;
33  
34  public class KotlinDataClassGenerator extends AbstractKotlinGenerator {
35  
36      public KotlinDataClassGenerator(String project) {
37          super(project);
38      }
39  
40      @Override
41      public List<KotlinFile> getKotlinFiles() {
42          FullyQualifiedTable table = introspectedTable.getFullyQualifiedTable();
43          progressCallback.startTask(getString("Progress.8", table.toString())); //$NON-NLS-1$
44          CommentGenerator commentGenerator = context.getCommentGenerator();
45          FullyQualifiedKotlinType type = new FullyQualifiedKotlinType(
46                  introspectedTable.getKotlinRecordType());
47  
48          KotlinFile kf = new KotlinFile(type.getShortNameWithoutTypeArguments());
49          kf.setPackage(type.getPackageName());
50  
51          KotlinType dataClass = KotlinType.newClass(type.getShortNameWithoutTypeArguments())
52                  .withModifier(KotlinModifier.DATA)
53                  .build();
54          kf.addNamedItem(dataClass);
55  
56          commentGenerator.addFileComment(kf);
57  
58          commentGenerator.addModelClassComment(dataClass, introspectedTable);
59  
60          List<IntrospectedColumn> introspectedColumns = introspectedTable.getAllColumns();
61  
62          for (IntrospectedColumn introspectedColumn : introspectedColumns) {
63              FullyQualifiedKotlinType kotlinType =
64                      JavaToKotlinTypeConverter.convert(introspectedColumn.getFullyQualifiedJavaType());
65  
66              KotlinProperty kp = KotlinProperty.newVar(introspectedColumn.getJavaProperty())
67                      .withDataType(kotlinType.getShortNameWithTypeArguments() + "?") //$NON-NLS-1$
68                      .withInitializationString("null") //$NON-NLS-1$
69                      .build();
70  
71              dataClass.addConstructorProperty(kp);
72  
73              kf.addImports(kotlinType.getImportList());
74          }
75  
76          if (context.getPlugins().kotlinDataClassGenerated(kf, dataClass, introspectedTable)) {
77              return listOf(kf);
78          } else {
79              return Collections.emptyList();
80          }
81      }
82  }
