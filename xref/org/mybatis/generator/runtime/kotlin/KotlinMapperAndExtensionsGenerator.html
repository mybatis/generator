71      protected void preCalculate() {
72          supportClassGenerator = new KotlinDynamicSqlSupportClassGenerator(context, introspectedTable, warnings);
73          recordType = new FullyQualifiedKotlinType(introspectedTable.getKotlinRecordType());
74          resultMapId = recordType.getShortNameWithoutTypeArguments() + "Result"; //$NON-NLS-1$
75          fragmentGenerator = new KotlinFragmentGenerator.Builder()
76                  .withIntrospectedTable(introspectedTable)
77                  .withResultMapId(resultMapId)
78                  .withDynamicSqlSupportClassGenerator(supportClassGenerator)
79                  .withTableFieldName(supportClassGenerator.getTablePropertyName())
80                  .build();
81  
82          hasGeneratedKeys = introspectedTable.getGeneratedKey().isPresent();
83      }
84  
85      protected KotlinFile createMapperInterfaceFile() {
86          FullyQualifiedKotlinType type = new FullyQualifiedKotlinType(
87                  introspectedTable.getMyBatis3JavaMapperType());
88  
89          KotlinFile kf = new KotlinFile(type.getShortNameWithoutTypeArguments());
90          kf.setPackage(type.getPackageName());
91  
92          return kf;
93      }
94  
95      protected KotlinType createMapperInterface(KotlinFile kotlinFile) {
96          FullyQualifiedKotlinType type = new FullyQualifiedKotlinType(
97                  introspectedTable.getMyBatis3JavaMapperType());
98  
99          KotlinType intf = KotlinType.newInterface(type.getShortNameWithoutTypeArguments())
100                 .withAnnotation("@Mapper") //$NON-NLS-1$
101                 .build();
102 
103         kotlinFile.addImport("org.apache.ibatis.annotations.Mapper"); //$NON-NLS-1$
104         kotlinFile.addNamedItem(intf);
105 
106         context.getCommentGenerator().addFileComment(kotlinFile);
107 
108         return intf;
109     }

136     protected boolean generate(KotlinFile kotlinFile, AbstractKotlinFunctionGenerator generator) {
137         KotlinFunctionAndImports mi = generator.generateMethodAndImports();
138         if (mi != null && generator.callPlugins(mi.getFunction(), kotlinFile)) {
139             kotlinFile.addNamedItem(mi.getFunction());
140             kotlinFile.addImports(mi.getImports());
141             return true;
142         }
143         return false;
144     }

146     protected boolean generate(KotlinFile kotlinFile, KotlinType kotlinType,
147                                AbstractKotlinFunctionGenerator generator) {
148         KotlinFunctionAndImports mi = generator.generateMethodAndImports();
149         if (mi != null && generator.callPlugins(mi.getFunction(), kotlinFile)) {
150             kotlinType.addNamedItem(mi.getFunction());
151             kotlinFile.addImports(mi.getImports());
152             return true;
153         }
154         return false;
155     }

157     @Override
158     public List<KotlinFile> getKotlinFiles() {
159         progressCallback.startTask(getString("Progress.17", //$NON-NLS-1$
160                 introspectedTable.getFullyQualifiedTable().toString()));
161         preCalculate();
162 
163         KotlinFile mapperFile = createMapperInterfaceFile();
164         KotlinType mapper = createMapperInterface(mapperFile);
165 
166         if (hasGeneratedKeys) {
167             addBasicInsertMethod(mapperFile, mapper);
168             addBasicInsertMultipleMethod(mapperFile, mapper);
169         }
170 
171         boolean reuseResultMap = addBasicSelectManyMethod(mapperFile, mapper);
172         addBasicSelectOneMethod(mapperFile, mapper, reuseResultMap);
173 
174         String mapperName = mapper.getName();
175 
176         addGeneralCountMethod(mapperFile, mapper, mapperName);
177         addGeneralDeleteMethod(mapperFile, mapper, mapperName);
178         addDeleteByPrimaryKeyMethod(mapperFile, mapperName);
179         addInsertOneMethod(mapperFile, mapper, mapperName);
180         addInsertMultipleMethod(mapperFile, mapper, mapperName);
181         addInsertMultipleVarargMethod(mapperFile, mapperName);
182         addInsertSelectiveMethod(mapperFile, mapper, mapperName);
183         addColumnListProperty(mapperFile);
184         addGeneralSelectMethod(mapperFile, mapperName);
185         addSelectDistinctMethod(mapperFile, mapperName);
186         addSelectByPrimaryKeyMethod(mapperFile, mapperName);
187         addGeneralUpdateMethod(mapperFile, mapper, mapperName);
188         addUpdateAllMethod(mapperFile);
189         addUpdateSelectiveMethod(mapperFile);
190         addUpdateByPrimaryKeyMethod(mapperFile, mapperName);
191         addUpdateByPrimaryKeySelectiveMethod(mapperFile, mapperName);
192 
193         KotlinFile supportFile = supportClassGenerator.getKotlinFile();
194 
195         List<KotlinFile> answer = new ArrayList<>();
196         if (context.getPlugins().dynamicSqlSupportGenerated(supportFile,
197                 supportClassGenerator.getOuterObject(),
198                 supportClassGenerator.getInnerClass(),
199                 introspectedTable)) {
200             answer.add(supportFile);
201         }
202 
203         if (context.getPlugins().mapperGenerated(mapperFile, mapper, introspectedTable)) {
204             answer.add(mapperFile);
205         }
206 
207         return answer;
208     }

210     protected void addInsertOneMethod(KotlinFile mapperFile, KotlinType mapper, String mapperName) {
211         InsertMethodGenerator generator = new InsertMethodGenerator.Builder()
212                 .withContext(context)
213                 .withIntrospectedTable(introspectedTable)
214                 .withTableFieldName(supportClassGenerator.getTablePropertyName())
215                 .withRecordType(recordType)
216                 .withMapperName(mapperName)
217                 .withSupportObjectImport(supportClassGenerator.getSupportObjectImport())
218                 .build();
219 
220         if (generate(mapperFile, generator) && !hasGeneratedKeys) {
221             addCommonInsertInterface(mapperFile, mapper);
222         }
223     }

228         mapperFile.addImport("org.mybatis.dynamic.sql.util.mybatis3.CommonInsertMapper"); //$NON-NLS-1$
229         mapperFile.addImports(recordType.getImportList());
230     }
