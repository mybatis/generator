16  package org.mybatis.generator.runtime.mybatis3;
17  
18  import static org.mybatis.generator.internal.util.messages.Messages.getString;
19  
20  import org.mybatis.generator.config.ClientGeneratorConfiguration;
21  import org.mybatis.generator.exception.InternalException;
22  import org.mybatis.generator.runtime.common.RecordModelGenerator;
23  import org.mybatis.generator.runtime.common.SimpleModelGenerator;
24  import org.mybatis.generator.runtime.mybatis3.javamapper.SimpleAnnotatedMapperGenerator;
25  import org.mybatis.generator.runtime.mybatis3.javamapper.SimpleJavaMapperGenerator;
26  import org.mybatis.generator.runtime.mybatis3.xmlmapper.SimpleXMLMapperGenerator;
27  
28  /**
29   * Runtime for generating simple MyBatis3 artifacts.
30   *
31   * <p>Simple means no "by example" methods, flat model, etc.
32   *
33   * @author Jeff Butler
34   */
35  public class LegacySimpleJavaRuntime extends LegacyJavaRuntime {
36      protected LegacySimpleJavaRuntime(Builder builder) {
37          super(builder);
38      }
39  
40      @Override
41      protected void calculateXmlMapperGenerator() {
42          if (context.getSqlMapGeneratorConfiguration().isPresent()) {
43              xmlMapperGenerator = initializeSubBuilder(new SimpleXMLMapperGenerator.Builder()).build();
44          }
45      }
46  
47      @Override
48      protected void calculateClientGenerator(ClientGeneratorConfiguration clientGeneratorConfiguration) {
49          clientGeneratorConfiguration.getLegacyClientType()
50                  .map(s -> switch (s) {
51                  case XML_MAPPER -> new SimpleJavaMapperGenerator.Builder();
52                  case ANNOTATED_MAPPER -> new SimpleAnnotatedMapperGenerator.Builder();
53                  default -> throw new InternalException(
54                          getString("ValidationError.31", s.name(), context.getId())); //$NON-NLS-1$
55                  })
56                  .ifPresent(builder -> {
57                      initializeSubBuilder(builder)
58                              .withProject(clientGeneratorConfiguration.getTargetProject())
59                              .build();
60                      javaGenerators.add(builder.build());
61                  });
62      }
63  
64      @Override
65      protected void calculateJavaModelGenerators() {
66          if (introspectedTable.isRecordBased()) {
67              var javaGenerator =  initializeSubBuilder(new RecordModelGenerator.Builder())
68                      .withProject(getModelProject())
69                      .build();
70              javaGenerators.add(javaGenerator);
71          } else {
72              var javaGenerator = initializeSubBuilder(new SimpleModelGenerator.Builder())
73                      .withProject(getModelProject())
74                      .build();
75              javaGenerators.add(javaGenerator);
76          }
77      }
78  
79      public static class Builder extends LegacyJavaRuntime.Builder {
80          @Override
81          protected Builder getThis() {
82              return this;
83          }
84  
85          @Override
86          public LegacySimpleJavaRuntime build() {
87              return new LegacySimpleJavaRuntime(this);
88          }
89      }
90  }
